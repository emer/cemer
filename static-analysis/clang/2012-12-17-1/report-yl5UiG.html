143void DumpPathSubList::AddPath(TypeDef* td, taBase* par, StringtaString& o, StringtaString& n) {
144 int o_i = o.length()-1;
145 int n_i = n.length()-1;
146 int o_last_sep = -1; // position of last separator character
147 int n_last_sep = -1;
148 // remove any trailing parts of name that are common to both..
149 // often only the leading part of name is new, so no need to
150 // replicate the same fix over and over..
151 while ((o_i >= 0) && (n_i >= 0) && (o[o_i] == n[n_i])) {
152 if(o[o_i] == '.') {
153 o_last_sep = o_i;
154 n_last_sep = n_i;
155 }
156 o_i--; n_i--;
157 }
158 StringtaString* op = (StringtaString*)&o;
159 StringtaString* np = (StringtaString*)&n;
160 StringtaString ot, nt;
161 if(o_last_sep > 0) {
162 op = &ot;
163 np = &nt;
164 ot = o.before(o_last_sep);
165 nt = n.before(n_last_sep);
166 }
167 if(taMisc::verbose_load >= taMisc::MESSAGES) {
168 StringtaString ppath = par->GetPathNames();
169 taMisc::Warning("---> New Path Fix, old:",*op,"new:",*np,"in:",ppath);
170 }
171 DumpPathSub* nwsb = new DumpPathSub(td, par, *op, *np);
172 Add(nwsb);
173 if(par != tabMisc::root) { // if local path, then add a global path fix too
174 StringtaString ppath = par->GetPath();
175 unFixPath(td, tabMisc::root, ppath); // un fix the parent if necessary
176 StringtaString long_o = ppath + *op;
177 StringtaString long_n = ppath + *np;
178 nwsb = new DumpPathSub(td, tabMisc::root, long_o, long_n);
179 Add(nwsb);
180// if(taMisc::verbose_load >= taMisc::MESSAGES) {
181 taMisc::Warning("---> New Global Path Fix, old:",long_o,"new:",long_n);
182// }
183 }

305taBase* DumpPathTokenList::FindFromPath(StringtaString& pat, TypeDef* td, void* base,
306 void* par, MemberDef* memb_def)
307{
308
309 if(pat[0] == '$') { // its a token path, not a real one..
310 StringtaString pat_act = pat(1, pat.length()-2); // actual path = nuke $ $'s
311 DumpPathToken* tok = NULL__null;
312 if(taMisc::strm_ver < 3) {
313 tok = SafeEl((int)pat_act); // just index of token item
314 if(tok == NULL__null) {
315 taMisc::Warning("Path Token Not Created Yet:", pat_act);
316 return NULL__null;
317 }
318 if((tok->object == NULL__null) && (base != NULL__null)) {
319 dumpMisc::vpus.AddVPU(base, (taBase*)par, pat_act, memb_def); // saving actual path here
320 return NULL__null;
321 }
322 return tok->object; // this is what we're looking for
323 }
324 else {
325 tok = FindName(pat_act); // using hash coded name lookup on raw path itself!
326 if(tok) {
327 if(tok->object)
328 return tok->object;
329 else {
330 if(base) {
331 dumpMisc::vpus.AddVPU(base, (taBase*)par, pat_act, memb_def);
332 }
333 }
334 return NULL__null;
335 }
336 // note: strm_ver >=3 with a $path$ without a tok will fall through to make a new token..
337 pat = pat_act + "$$"; // just trun it into a simple definition version
338 }
339 }
340
341 // target type and name, if present -- to assist in finding objects if not found
342 TypeDef* trg_td = NULL__null;
343 StringtaString trg_nm;
344
345 DumpPathToken* tok = NULL__null;
346 int tok_idx = -1;
347 if(pat.lastchar() == '$'|| pat.lastchar() == '>') { // path contains definition of new token
348 StringtaString token_id = pat.after('$');
349 token_id = token_id.before('$');
350 if(pat.lastchar() == '>') { // name, type hint
351 StringtaString typnm = pat.between('<', '>');
352 trg_nm = typnm.after(',');
353 typnm = typnm.before(',');
354 trg_td = taMisc::FindTypeName(typnm);
355 }
356 pat = pat.before('$');
357 if(token_id.empty()) // version 3 = its just the path
358 token_id = pat;
359 tok = NewLoadToken(pat, token_id); // make the token
360 tok_idx = size-1;
361 }
362
363 dumpMisc::path_subs.FixPath(td, tabMisc::root, pat);
364 MemberDef* md = NULL__null;
365 taBase* rval = tabMisc::root->FindFromPath(pat, md);
366 if(rval && md && (md->type->ptr == 1)) { // deref ptr
367 rval = *((taBase**)rval);
368 }
369 rval = FixPathFind(pat, rval, trg_td, trg_nm); // try to fix based on trg info
370 if(!rval) {
371 if(base != NULL__null)
372 dumpMisc::vpus.AddVPU(base, (taBase*)par, pat, memb_def);
373 return NULL__null;
374 }
375 if(tok) {
376 if(!(tok->object) && rval) { // we're going to set this guy -- add to hash table!
377 obj_hash_table.AddHashPtr(rval, tok_idx); // only case for this is last guy
378 }
379 tok->object = rval;
380 }
381 return rval;
382}

384taBase* DumpPathTokenList::FixPathFind(const StringtaString& pat, taBase* init_find,
385 TypeDef* trg_typ, const StringtaString& trg_nm) {
386 if(!trg_typ) return init_find; // typ info should be on all cases if present at all
387 bool name_ok = true;
388 bool type_ok = true;
389 if(init_find) {
390 if(!init_find->InheritsFrom(trg_typ)) {
391 // not good -- must inherit from proper type!
392 type_ok = false;
393 }
394 if(trg_nm.nonempty() && (init_find->GetName() != trg_nm)) {
395 // see if we can find one with right name..
396 name_ok = false;
397 }
398 if(!name_ok || !type_ok) goto find_new;
399 return init_find; // must be good, go with it!
400 }
401
402 find_new: // find a new guy -- find a root to search from
403 taList_impl* find_root = NULL__null;
404 if(init_find && init_find->GetOwner()) {
405 // find any kind of list owner!
406 find_root = (taList_impl*)init_find->GetOwner(&TA_taList_impl);
407 }
408
409 if(!find_root) { // didn't get anything -- try walking up path to prev [
410 // the get owner thing should have worked but this is just a failsafe..
411 if(pat.contains('[')) {
412 StringtaString owpth = pat.before('[', -1);
413 MemberDef* md;
414 find_root = (taList_impl*)tabMisc::root->FindFromPath(owpth, md);
415 if(find_root && !find_root->InheritsFrom(&TA_taList_impl)) {
416 find_root = (taList_impl*)find_root->GetOwner(&TA_taList_impl);
417 }
418 }
419 }
420
421 if(!find_root) return init_find; // couldn't do anything better here.
422
423 taBase* rval = NULL__null;
424 if(find_root->InheritsFrom(&TA_taGroup_impl)) {
425 taGroup_impl* gprt = (taGroup_impl*)find_root;
426 if(gprt->root_gp)
427 gprt = gprt->root_gp; // go up to root of initial find owner
428 if(trg_nm.nonempty()) {
429 rval = gprt->FindLeafName_(trg_nm);
430 if(!rval && !type_ok) {
431 // didn't find name -- only if our orig did not have a good type, search for new type
432 rval = gprt->FindLeafType_(trg_typ);
433 }
434 }
435 else { // try type..
436 rval = gprt->FindLeafType_(trg_typ);
437 }
438 }
439 else {
440 // no leaf searches for plain lists..
441 if(trg_nm.nonempty()) {
442 rval = (taBase*)find_root->FindName_(trg_nm);
443 // don't worry about type here -- could have changed -- name is a good enough match
444 if(!rval && !type_ok) {
445 // didn't find name -- only if our orig did not have a good type, search for new type
446 rval = find_root->FindType_(trg_typ);
447 }
448 }
449 else { // try type
450 rval = find_root->FindType_(trg_typ);
451 }
452 }
453 if(rval) return rval; // if we got something, use it!
454 return init_find; // otherwise fall back on this
455}

493 if ((is_static && !HasOption("SAVE")) || HasOption("NO_SAVE"))
494 return false;
495 // if taBase, query it for member save
496 TypeDef* par_typ = GetOwnerType();
497 if (par && par_typ && par_typ->InheritsFrom(&TA_taBase)) {
498 taBase* par_ = (taBase*)par;
499 taBase::DumpQueryResult dqr = par_->Dump_QuerySaveMember(this);
500 if (dqr == taBase::DQR_NO_SAVE) return false;
501 else if (dqr == taBase::DQR_SAVE) return true;
502 // else default, so fall through
503 }
504
505 // first, check explicit rules
506 if (HasOption("SAVE"))
507 return true;
508 else if (HasOption("NO_SAVE"))
509 return false;
510 else if (HasOption("NO_SAVE_EMPTY")) {
511 void* new_base = GetOff(par);
512 return !(type->ValIsEmpty(new_base, this));
513 }
514 else if (HasOption("NO_SAVE_DEF")) {
515 // note: if no DEF then this will behave like NO_SAVE_EMPTY
516// void* new_base = GetOff(par);
517 return !(ValIsDefault(par, taMisc::SHOW_CHECK_MASK));
518 }
519 // embedded types (simple or objects) get saved by default
520 else if (type->ptr == 0)
521 return true;
522 // ok, so it is a ptr -- some types get saved by default
523 else if (type->DerivesFrom(TA_taBase) ||
524 type->DerivesFrom(TA_TypeDef) ||
525 type->DerivesFrom(TA_MemberDef) ||
526 type->DerivesFrom(TA_MethodDef))
527 return true;
528 else
529 // if its a pointer object you own
530 return false;

579 }
580 }
581
582 taMisc::indent(strm, indent, 1) << name;
583
584 bool save_value = true;
585 if (type->InheritsFrom(TA_Variant)) {
586 Variant& var = *((Variant*)(new_base));
587 var.Dump_Save_Type(strm);
588 // we don't try to save null atomics

590 save_value = false;
591 }
592 }

601 }

605}

923 const char* prv_read_nm, int prv_c) {
924 if(taMisc::verbose_load >= taMisc::TRACE) {
925 const char* nm = (prv_read_nm != NULL__null) ? prv_read_nm : "NULL";
926 StringtaString msg;
927 msg << "Entering MemberSpace::Dump_Load, type: " << owner->name
928 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = " << StringtaString((ta_intptr_t)base)
929 << ", prv_read_nm = " << nm
930 << ", prv_c = " << prv_c;
931 taMisc::Info(msg);
932 }
933
934 int rval = 2; // default assumption is that no members are loaded
935 do {
936 int c;
937 if(prv_read_nm != NULL__null) {
938 taMisc::LexBuf = prv_read_nm;
939 c = prv_c;
940 prv_read_nm = NULL__null;
941 }
942 else {
943 c = taMisc::read_word(strm, true);
944 }
945 if(c == EOF(-1)) {
946 if(taMisc::verbose_load >= taMisc::MESSAGES)
947 taMisc::Warning("<<< EOF in MemberSpace::Dump_Load", taMisc::LexBuf);
948 return EOF(-1);
949 }
950 if(c == '}') {
951 strm.get(); // get the bracket (above was peek)
952 if(strm.peek() == ';') strm.get(); // skip past ending semi
953 if(taMisc::verbose_load >= taMisc::TRACE) {
954 const char* nm = (prv_read_nm != NULL__null) ? prv_read_nm : "NULL";
955 StringtaString msg;
956 msg << "}, Leaving MemberSpace::Dump_Load, type: " << owner->name
957 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = "
958 << StringtaString((ta_intptr_t)base) << ", prv_read_nm = " << nm
959 << ", prv_c = " << prv_c;
960 taMisc::Info(msg);
961 }
962 return rval;
963 }
964 if(c == '$') { // got a path token
965 strm.get();
966 c = taMisc::read_word(strm);
967 StringtaString tok_nm = taMisc::LexBuf;
968 strm.get(); // read the end of token
969 c = taMisc::skip_white(strm); // get the equals
Value stored to 'c' is never read
970 c = taMisc::read_till_lb_or_semi(strm);
971 if(c == EOF(-1)) {
972 if(taMisc::verbose_load >= taMisc::MESSAGES)
973 taMisc::Warning("<<< EOF in MemberSpace::Dump_Load", taMisc::LexBuf);
974 return EOF(-1);
975 }
976 StringtaString path_val = taMisc::LexBuf;
977 dumpMisc::path_tokens.NewLoadToken(path_val, tok_nm);
978 continue; // keep on truckin
979 }
980
981 StringtaString mb_name = taMisc::LexBuf;
982 if((c == ' ') || (c == '\n') || (c == '\t')) { // skip past white
983 strm.get();
984 c = taMisc::skip_white(strm, true);
985 }
986 int tmp;
987 if((c == '@') || (c == '.') || (c == '*')) { // got a path
988 // pass the mb_name, which is the type name, to the load function...
989 tmp = TA_taBase.Dump_Load_impl(strm, NULL__null, base, mb_name);
990 }
991 else if (c == '[') {
992 // finished with members, mat data coming up (so exit members)
993 return 3; // special kludgy code to tell Matrix::Dump_Load to expect mat vals
994 } else {
995 MemberDef* md = FindName(mb_name);
996 if(md == NULL__null) { // try to find a name with an aka..
997 int a;
998 for(a=0;a<size;a++) {
999 MemberDef* amd = FastEl(a);
1000 StringtaString aka = amd->OptionAfter("AKA_");
1001 if(aka.empty()) continue;
1002 if(aka == mb_name) {
1003 md = amd;
1004 break;
1005 }
1006 }
1007 }
1008 if(md != NULL__null) {
1009 // if md is a Variant, expect the type/null codes -- load them before proceeding
1010
1011 TypeDef* type = md->type;
1012 if (type->DerivesFrom(TA_Variant)) {
1013 Variant& var = *((Variant*)(md->GetOff(base)));
1014 // read vartype(int) and null (1/0):
1015 //NOTE: we don't actually bail if not found, to support possible case where
1016 // a non-variant member was changed to be a variant -- however, the loading
1017 // could fail because it may not be set to the the correct type
1018 if (!var.Dump_Load_Type(strm, c)) {
1019 taMisc::Warning("expected variant type information for member:", md->name,
1020 "in type:", md->GetOwnerType()->name);
1021 }
1022 }
1023 tmp = md->Dump_Load(strm, base, par);
1024 if(tmp == EOF(-1)) {
1025 if(taMisc::verbose_load >= taMisc::MESSAGES)
1026 taMisc::Warning("<<< EOF in MemberSpace::Dump_Load::else:", mb_name);
1027 return EOF(-1);
1028 }
1029 if(tmp == false)
1030 taMisc::skip_past_err(strm);
1031 }
1032 else {
1033 if(taMisc::verbose_load >= taMisc::VERSION_SKEW)
1034 taMisc::Warning("Member:",mb_name,"not found in type:",
1035 owner->name, "(this is likely just harmless version skew)");
1036 int sv_vld = taMisc::verbose_load;
1037 if(taMisc::verbose_load >= taMisc::VERSION_SKEW)
1038 taMisc::verbose_load = taMisc::SOURCE;
1039 taMisc::skip_past_err(strm);
1040 taMisc::verbose_load = (taMisc::LoadVerbosity)sv_vld;
1041 if(taMisc::verbose_load >= taMisc::VERSION_SKEW) {
1042 taMisc::Info("\n\n");
1043 }
1044 }
1045 rval = 1; // member was loaded, do update after edit
1046 }
1047 } while (1);
1048 if(taMisc::verbose_load >= taMisc::TRACE) {
1049 const char* nm = (prv_read_nm != NULL__null) ? prv_read_nm : "NULL";
1050 StringtaString msg;
1051 msg << "Leaving MemberSpace::Dump_Load, type: " << owner->name
1052 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = " << StringtaString((ta_intptr_t)base)
1053 << ", prv_read_nm = " << nm
1054 << ", prv_c = " << prv_c;
1055 taMisc::Info(msg);
1056 }
1057 return rval;
1058}

1077 int rval;

1091 << StringtaString((ta_intptr_t)base);
1092 taMisc::Info(msg);

1095 }

1126 taMisc::Info(msg);
1127 }
1128 //TODO: if we allow Variant taBase pointers, need to detect and do FixNull
1129 // if (type->InheritsFormal(TA_Variant) ... do FixNull
1130 return rval;
1131 }

1135 // if semi, no value is present, so just exit

1150
1151 } else {

1153 }
1154
1155 if (c != ';') {
1156 taMisc::Warning("Missing ';' in dump file for member:", name,

1159 }

1161 if(taMisc::verbose_load >= taMisc::TRACE) {
1162 StringtaString msg;
1163 msg << "Leaving MemberDef::Dump_Load, member: " << name
1164 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = "
1165 << StringtaString((ta_intptr_t)base);

1177 StringtaString msg;

1190 return EOF(-1);

1214 has_type = true; // actually do have type name
1215 }
1216 else {
1217 if(has_type) {
1218 tpnm = taMisc::LexBuf.before(spc_idx);
1219 path = taMisc::LexBuf.after(spc_idx);
1220 }
1221 else {
1222 path = taMisc::LexBuf;
1223 }
1224 }
1225 if(!has_type)

1269int TypeDef::Dump_Load_Path_impl(istream&, void*& base, void* par, StringtaString path) {
1270 bool ptr_flag = false; // pointer was loaded
1271 StringtaString orig_path;
1272 taBase* find_base = NULL__null; // where to find item from
1273 if(taMisc::verbose_load >= taMisc::TRACE) {
1274 StringtaString msg;
1275 msg << "Entering TypeDef::Dump_Load_Path_impl, type: " << name
1276 << ", path = " << path
1277 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = "
1278 << StringtaString((ta_intptr_t)base);
1279 taMisc::Info(msg);
1280 }
1281
1282 if(path.firstchar() == '@') { // relative to current parent..
1283 if(par == NULL__null) {
1284 taMisc::Warning("Dump_Load_path_impl: Relative path with NULL parent:", path);
1285 return false;
1286 }
1287 path = path.after('@');
1288 find_base = (taBase*)par; // we better get a taBase
1289 }
1290
1291 if(path.firstchar() == '*') {
1292 ptr_flag = true;
1293 path = path.after('*');
1294 }
1295
1296 // strip any leading parens, etc
1297 while(path.firstchar() == '(') path = path.after('(');
1298 while(path.lastchar() == ')') path = path.before(')',-1);
1299
1300 if(find_base==NULL__null)
1301 find_base = tabMisc::root; // search from top if not relative
1302
1303 dumpMisc::path_subs.FixPath(NULL__null, find_base, path); // fix the path name.. (substitution)
1304 orig_path = path; // original is post-substitution
1305
1306 if((path == "root") || (path == ".")) {
1307 base = (void*)find_base;
1308 if(taMisc::verbose_load >= taMisc::TRACE) {
1309 StringtaString msg;
1310 msg << "root Leaving TypeDef::Dump_Load_Path_impl, type: " << name
1311 << ", path = " << path
1312 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = "
1313 << StringtaString((ta_intptr_t)base);
1314 taMisc::Info(msg);
1315 }
1316 return true;
1317 }
1318
1319 int delim_pos = taBase::GetLastPathDelimPos(path);
1320 StringtaString el_path;
1321 if(path[delim_pos] == '[')
1322 el_path = path.from(delim_pos);
1323 else
1324 el_path = path.after(delim_pos);
1325 StringtaString ppar_path;
1326 if(delim_pos > 0)
1327 ppar_path = path.before(delim_pos);
1328
1329 MemberDef* ppar_md = NULL__null;
1330 taBase* ppar = find_base->FindFromPath(ppar_path, ppar_md); // path-parent
1331
1332 if(!ppar) {
1333 taMisc::Warning("Dump_Load_path_impl: Could not find a parent for:",el_path,"in",ppar_path);
1334 return false;
1335 }
1336
1337 if(ppar_md && !ppar_md->type->InheritsFrom(TA_taBase)) {
1338 taMisc::Warning("Dump_Load_path_impl: Parent must be a taBase type for:",el_path,"in",ppar_path,
1339 "type:",ppar_md->type->name);
1340 return false;
1341 }
1342
1343 // this is where the parent loads the child! target type is this type
1344 void* nw_base = NULL__null;
1345 if(ptr_flag)
1346 nw_base = ppar->Dump_Load_Path_ptr(el_path, this);
1347 else
1348 nw_base = ppar->Dump_Load_Path_parent(el_path, this);
1349
1350 if(nw_base) {
1351 base = nw_base;
1352 return true;
1353 }
1354 else {
1355 return false;
1356 }
1357}

1447int TypeDef::Dump_Load_impl(istream& strm, void* base, void* par, const char* typnm) {
1448 if(taMisc::verbose_load >= taMisc::TRACE) {
1449 const char* nm = (typnm != NULL__null) ? typnm : "NULL";
1450 StringtaString msg;
1451 msg << "Entering TypeDef::Dump_Load_impl, type: " << owner->name
1452 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = "
1453 << StringtaString((ta_intptr_t)base) << ", typnm = " << nm;
1454 taMisc::Info(msg);
1455 }
1456 TypeDef* td = NULL__null;
1457 StringtaString path;
1458 int rval = Dump_Load_Path(strm, base, par, td, path, typnm);
1459 if((taMisc::verbose_load >= taMisc::TRACE) && (td != NULL__null)) {
1460 StringtaString msg;
1461 msg << "Loading: " << td->name << " " << path << " rval: " << rval;
1462 taMisc::Info(msg);
1463 }
1464 if((rval > 0) && (base != NULL__null)) {
1465 if(td->InheritsFrom(TA_taBase)) {
1466 taBase* rbase = (taBase*)base;
1467 rval = rbase->Dump_Load_Value(strm, (taBase*)par);
1468 if(rval==1) {
1469 if (rbase->HasOption("IMMEDIATE_UPDATE"))
1470 rbase->UpdateAfterEdit();
1471 else if(!rbase->HasOption("NO_UPDATE_AFTER")) {
1472 dumpMisc::update_after.Link(rbase);

1476 dumpMisc::post_update_after.Link(rbase);
1477 }
1478 }
1479 }
1480 else {
1481 rval = td->Dump_Load_Value(strm, base, par);
1482 }

1484 if(taMisc::verbose_load >= taMisc::MESSAGES)
1485 taMisc::Warning("<<< EOF in Dump_Load_impl:", name);
1486 return EOF(-1);
1487 }
1488 if(rval == false) {
1489 // read till next rbracket, (skipping over inner ones..)
1490 // be sure to revise dump_loads to return false on error

1494 const char* nm = (typnm != NULL__null) ? typnm : "NULL";

1498 << StringtaString((ta_intptr_t)base) << ", typnm = " << nm;
1499 taMisc::Info(msg);
1500 }
1501 return true; // we already scanned past error
1502 }
1503 }
1504 else {
1505 // make sure we skip until we hit the end of this one (rbracket)
1506 // not just the semicolon at the end of some data item..
1507 if(strm.peek() == '{')
1508 strm.get(); // if we're at the start of a bracket, get it first
1509 if(strm.peek() == '=') { // if we're reading a path and it didn't work, need to get lb!
1510 taMisc::read_till_lbracket(strm, false);
1511 }
1512 int c;
1513 // if((path.length() > 1) && (path[1] == '*')) // for path items, use non-peek, else use peek
1514 // cerr << "***HERE***" << endl;
1515 c = taMisc::skip_past_err_rb(strm, false);
1516// else
1517// c = taMisc::skip_past_err_rb(strm, true);
1518 if(c == EOF(-1)) return EOF(-1);
1519 if(taMisc::verbose_load >= taMisc::TRACE) {
1520 const char* nm = (typnm != NULL__null) ? typnm : "NULL";
1521 StringtaString msg;
1522 msg << "err rb Leaving TypeDef::Dump_Load_impl, type: " << owner->name
1523 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = "
1524 << StringtaString((ta_intptr_t)base) << ", typnm = " << nm;
1525 taMisc::Info(msg);
1526 }
1527 return true; // already scanned past error
1528 }
1529 if(taMisc::verbose_load >= taMisc::TRACE) {
1530 const char* nm = (typnm != NULL__null) ? typnm : "NULL";
1531 StringtaString msg;
1532 msg << "Leaving TypeDef::Dump_Load_impl, type: " << owner->name
1533 << ", par = " << StringtaString((ta_intptr_t)par) << ", base = "
1534 << StringtaString((ta_intptr_t)base) << ", typnm = " << nm
1535 << ", rval = " << rval;
1536 taMisc::Info(msg);
1537 }
1538 return rval;
1539}

1541int TypeDef::Dump_Load(istream& strm, void* base, void* par, void** el_) {
1542//WARNING: DO NOT put any calls to eventloop in the load code -- it will cause crashes
1543 if (el_) *el_ = NULL__null; //default if error
1544 if(base == NULL__null) {
1545 taMisc::Warning("Cannot load into NULL");
1546 return false;
1547 }
1548
1549 dumpMisc::path_subs.Reset();
1550 dumpMisc::path_tokens.ReInit();
1551 dumpMisc::vpus.Reset();
1552 dumpMisc::update_after.Reset();
1553 // NOTE: post_update_after can include load commands so don't reset here
1554 // -- is auto reset after running!
1555 // dumpMisc::post_update_after.Reset();
1556 tabMisc::root->plugin_deps.Reset();
1557
1558 int c;
1559 c = taMisc::read_till_eol(strm);
1560 if(c == EOF(-1)) return EOF(-1);
1561 if (taMisc::LexBuf.contains("// ta_Dump File v1.0")) {
1562 taMisc::strm_ver = 1;
1563 taMisc::loading_version.set(3,2,0); // old pdp++ guy
1564 }
1565 else if(taMisc::LexBuf.contains("// ta_Dump File v2.0")) {
1566 taMisc::strm_ver = 2;
1567 }
1568 else if(taMisc::LexBuf.contains("// ta_Dump File v3.0")) {
1569 taMisc::strm_ver = 3;
1570 }
1571 else {
1572 taMisc::Warning("Dump file does not have proper format id:", taMisc::LexBuf);
1573 return false;
1574 }
1575 if(taMisc::strm_ver >= 2) {
1576 if(taMisc::LexBuf.contains(" -- code v")) { // code version stamped into file
1577 StringtaString ldver = taMisc::LexBuf.after(" -- code v");
1578 taMisc::loading_version.setFromString(ldver);
1579 }
1580 else {
1581 taMisc::loading_version.set(4,0,19); // last version without explicit versioning
1582 }
1583 }
1584
1585 if(taMisc::loading_version > taMisc::version_bin) {
1586 taMisc::Warning("Loading a file saved in a *LATER* version of the software:",
1587 taMisc::loading_version.toString()," Current version is:",
1588 taMisc::version_bin.toString(),
1589 "this is not likely to end well, as the software is generally only backwards compatible, but it might work..");
1590 }
1591
1592 TypeDef* td;
1593 StringtaString path;
1594 int rval = Dump_Load_Path(strm, base, par, td, path); // non-null base just gets type
1595 if(rval <= 0) {
1596 taMisc::Warning("Dump load aborted due to errors");
1597 return false;
1598 }
1599
1600 if(!td->InheritsFrom(TA_taBase)) {
1601 taMisc::Warning("Only taBase objects may be loaded, not:", td->name);
1602 return false;
1603 }
1604
1605 if(taMisc::verbose_load >= taMisc::TRACE) {
1606 cout << "Loading: " << td->name << " " << path << " rval: " << rval << "\n";
1607 }
1608
1609 // locals needed below, so we can use a jump
1610 StringtaString new_path;
1611 taBase* el = NULL__null; // the loaded element
1612
1613 ++taMisc::is_loading;
1614
1615 // check for plugin deps, if so, load those in and resume
1616 if (td->InheritsFrom(TA_taPluginBase_List) &&
1617 (path == ".plugin_deps"))
1618 {
1619 rval = tabMisc::root->Dump_Load_Value(strm, NULL__null); // read it
1620 rval = Dump_Load_impl(strm, NULL__null, tabMisc::root); // base is given by the path..
1621 // resolve plugin dependencies, and warn/abort if not all are present
1622 if (!tabMisc::root->VerifyHasPlugins()) {
1623 rval = false;
1624 goto endload;
1625 }
1626
1627 //read again -- get the actual path
1628 rval = Dump_Load_Path(strm, base, par, td, path); // non-null base just gets type
1629 }
1630
1631
1632 if(InheritsFrom(td)) { // we are the same as load token
1633 el = (taBase*)base; // so use the given base
1634 }
1635 else {
1636 taBase* par = (taBase*)base; // given base must be a parent
1637 el = par->New(1,td); // create one of the saved type
1638 if(el == NULL__null) {
1639 taMisc::Warning("Could not make a:",td->name,"in:",par->GetPathNames());
1640 rval = false;
1641 goto endload;
1642 }
1643 }
1644
1645 if(path.contains('\"')) {
1646 StringtaString elnm = path.before('\"',-1);
1647 elnm = elnm.after('\"',-1);
1648 if(elnm.nonempty()) {
1649 el->SetName(elnm);
1650 }
1651 new_path = el->GetPath(); // for head item, must be regular path in case loading 2 of same name!
1652 }
1653 else {
1654 new_path = el->GetPath();
1655 }
1656
1657 if(new_path != path)
1658 dumpMisc::path_subs.AddPath(td, tabMisc::root, path, new_path);
1659
1660 dumpMisc::dump_root = el;
1661 dumpMisc::dump_root_path = new_path;
1662
1663 rval = el->Dump_Load_Value(strm, (taBase*)par); // read it
1664
1665 while(rval != EOF(-1)) {
1666 rval = Dump_Load_impl(strm, NULL__null, par); // base is given by the path..
1667 }
1668
1669 dumpMisc::vpus.Resolve(); // try to cache out references.
1670
1671 for (int i=0; i<dumpMisc::update_after.size; i++) {
1672 taBase* tmp = dumpMisc::update_after.FastEl(i);
1673 if(taBase::GetRefn(tmp) <= 1) {
1674 taMisc::Warning("Object: of type:",
1675 tmp->GetTypeDef()->name,"named:",tmp->GetName(),"is unowned!");
1676 taBase::Ref(tmp);
1677 }
1678 tmp->UpdateAfterEdit();
1679 }
1680 rval = true;
1681
1682endload:
1683 --taMisc::is_loading;
1684
1685 dumpMisc::update_after.Reset(); //note: don't reset post list!
1686 dumpMisc::path_subs.Reset();
1687 dumpMisc::path_tokens.Reset();
1688 dumpMisc::vpus.Reset();
1689 if (el_) *el_ = (void*)el;
1690 // if there were any post guys, send a msg to the taiMisc object, who will call us
1691 // back when the eventloop gets processed next
1692 if (dumpMisc::post_update_after.size > 0) {
1693 if (!taMisc::is_post_loading) {
1694 if(taMisc::is_undo_loading) {
1695 // do them right away, so undo_loading is still present -- shouldn't be a prob
1696 // because everything is already there..
1697 dumpMisc::PostUpdateAfter();
1698 }
1699 else {
1700 QTimer::singleShot(0, taiMC_, SLOT(PostUpdateAfter())"1""PostUpdateAfter()" );
1701 }
1702 } else { // shouldn't happen???
1703 dumpMisc::post_update_after.Reset(); // if it does happen, don't leave around
1704 }
1705 }
1706
1707 if(taMisc::loading_version < taMisc::version_bin) {
1708 if(td && td->InheritsFrom(&TA_taProject) && taMisc::gui_active) {
1709 // only for project-level objects, not sub-files, and only in gui mode (so analysis scripts and startup things don't give a lot of grief
1710 taMisc::Warning("Loaded a file saved in an earlier version of the software:",
1711 taMisc::loading_version.toString()," Current version is:",
1712 taMisc::version_bin.toString(),
1713 "set verbose_load to VERSION_SKEW to see version skew warnings, which may have useful information -- see the current ChangeLog info on the emergent wiki for things you should be looking for");
1714 }
1715 }
1716 else if(taMisc::loading_version > taMisc::version_bin) {
1717 taMisc::Warning("Loaded a file saved in a *LATER* version of the software:",
1718 taMisc::loading_version.toString()," Current version is:",
1719 taMisc::version_bin.toString(),
1720 "this is not likely to be a good idea, as the software is generally only backwards compatible, but it might work..");
1721 }
1722 return rval;
1723}
