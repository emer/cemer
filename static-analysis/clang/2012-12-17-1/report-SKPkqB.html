713 }
714
715 if (!limitedDrag && (pos < min || pos > max)) {
716 qreal pixPerUnit = ((max - min) * 2.0 * transmissionRatio) / (q->maximum() - q->minimum());
717 if (pos < min)
718 return qRound(q->minimum() - (min-pos) / pixPerUnit);
719 else
720 return qRound((q->maximum() / (2.0 * transmissionRatio)) + (pos-max) / pixPerUnit);
721 }
722
723 double alpha;
724 if (pos < min)
725 alpha = 0;
726 else if (pos > max)
727 alpha = m_pi;
728 else
729 alpha = acos(1.0 - 2.0 * double(pos - min) / double(max - min));// ### taylor
730
731 double deg = alpha * rad_factor / transmissionRatio;
732 return q->minimum() + int(double(q->maximum() - q->minimum()) * deg / 360.0);
733}
