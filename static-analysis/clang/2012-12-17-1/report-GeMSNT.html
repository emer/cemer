831cssEl* cssEl::GetElFromTA(TypeDef* td, void* itm, const StringtaString& nm, MemberDef* md,
832 cssEl* class_parent) {
833 TypeDef* nptd = td->GetNonPtrType(); // always create one of these
834
835 if(!nptd)
836 return &cssMisc::Void;
837
838 bool ro = md && md->HasOption("READ_ONLY");
839
840 int new_ptr = td->ptr;
841 if(new_ptr == 1) new_ptr++; // bump up to a **
842 else if(new_ptr > 1) {
843 cssMisc::Error(NULL__null, "GetElFromTA -- cannot process ** (pointer-pointer) C members!");
844 return &cssMisc::Void;
845 }
846
847 if(nptd->InheritsNonAtomicClass()) {
848 if(nptd->DerivesFrom(&TA_ios) || nptd->DerivesFrom(&TA_istream)
849 || nptd->DerivesFrom(&TA_ostream) || nptd->DerivesFrom(&TA_iostream))
850 return new cssIOS(itm, new_ptr, nptd, nm, class_parent, ro);
851 else if(nptd->DerivesFrom(TA_taMatrix))
852 return new cssTA_Matrix(itm, new_ptr, nptd, nm, class_parent, ro);
853 else if(nptd->DerivesFrom(TA_taBase))
854 return new cssTA_Base(itm, new_ptr, nptd, nm, class_parent, ro);
855 else
856 return new cssTA(itm, new_ptr, nptd, nm, class_parent, ro);
857 }
858 else {
859 if(nptd->InheritsFormal(TA_class)) {
860 if(nptd == &TA_taString)
861 return new cssCPtr_String(itm, new_ptr, nm, class_parent, ro);
862 else if(nptd == &TA_Variant)
863 return new cssCPtr_Variant(itm, new_ptr, nm, class_parent, ro);
864 else if(nptd == &TA_TypeDef)
865 return new cssTypeDef(itm, new_ptr, nptd, nm, class_parent, ro);
866 else if(nptd == &TA_MemberDef)
867 return new cssMemberDef(itm, new_ptr, nptd, nm, class_parent, ro);
868 else if(nptd == &TA_MethodDef)
869 return new cssMethodDef(itm, new_ptr, nptd, nm, class_parent, ro);
870 else if(nptd->DerivesFrom(TA_taSmartRef))
871 return new cssSmartRef(itm, new_ptr, nptd, nm, class_parent, ro);
872 else
873 return new cssTA(itm, new_ptr, nptd, nm, class_parent, ro);
874 }
875 else {
876 if(nptd->DerivesFrom(TA_bool))
877 return new cssCPtr_bool(itm, new_ptr, nm, class_parent, ro);
878 else if(nptd->DerivesFormal(TA_enum))
879 return new cssCPtr_enum(itm, new_ptr, nm, class_parent, ro);
880 else if ((nptd->DerivesFrom(TA_int) || nptd->DerivesFrom(TA_unsigned_int)))
881 return new cssCPtr_int(itm, new_ptr, nm, class_parent, ro);
882 else if(nptd->DerivesFrom(TA_short) || (nptd->DerivesFrom(TA_unsigned_short)))
883 return new cssCPtr_short(itm, new_ptr, nm, class_parent, ro);
884 else if (nptd->DerivesFrom(TA_long) || nptd->DerivesFrom(TA_unsigned_long))
885 return new cssCPtr_long(itm, new_ptr, nm, class_parent, ro);
886 else if (nptd->DerivesFrom(TA_char) || nptd->DerivesFrom(TA_unsigned_char)
887 || nptd->DerivesFrom(TA_signed_char))
888 return new cssCPtr_char(itm, new_ptr, nm, class_parent, ro);
889 else if(nptd->DerivesFrom(TA_int64_t) || nptd->DerivesFrom(TA_uint64_t))
890 return new cssCPtr_long_long(itm, new_ptr, nm, class_parent, ro);
891 else if(nptd->DerivesFrom(TA_float))
892 return new cssCPtr_float(itm, new_ptr, nm, class_parent, ro);
893 else if(nptd->DerivesFrom(TA_double))
894 return new cssCPtr_double(itm, new_ptr, nm, class_parent, ro);
895 else {
896 cssMisc::Error(NULL__null, "GetElFromTA -- atomic class not managed in case:", nptd->name);
897 return &cssMisc::Void;
898 }
899 }
900 }

1119
1120void cssElFun::BindArgs(cssEl** args, int& act_argc) {
1121 args[0] = this; // first argument is always selfptr
1122 act_argc = 0;
1123 if(argc == NoArg) return;
1124
1125 cssSpace* stack = prog->Stack();
1126 int stack_start; // where to start getting things off of the stack
1127
1128 if(argc == 0) {
1129 if(stack->Peek() == &cssMisc::Void) // get rid of arg stop..
1130 stack->Pop();
1131 return;
1132 }
1133
1134 if(argc == VarArg) {
1135 ///////////////////////////////////////////////////////////////////////////
1136 // variable numbers of arguments
1137 for(stack_start = stack->size-1; stack_start >= 0; stack_start--) {
1138 if(stack->FastEl(stack_start) == &cssMisc::Void)
1139 break;
1140 }
1141 act_argc = ((int)stack->size - stack_start) - 1;
1142 if(act_argc >= ArgMax) {
1143 cssMisc::Error(prog, "Arg count greater than max (32) in:", (const char*)name,
1144 StringtaString((int)act_argc));
1145 act_argc = -1;
1146 return;
1147 }
1148 for(int i=act_argc; i>0; i--)
1149 args[i] = stack->Pop();
1150
1151 if(stack->Peek() == &cssMisc::Void) // get rid of arg stop..
1152 stack->Pop();
1153 }
1154 else if(arg_defs.size == 0) {
1155 ///////////////////////////////////////////////////////////////////////////
1156 // no default args -- must have correct number!
1157
1158 if(stack->size < argc) {
1159 cssMisc::Error(prog, "Incomplete argument list for:", (const char*)name,
1160 "should have at least:", StringtaString((int)argc), "got:",
1161 StringtaString((int)stack->size));
1162 act_argc = -1;
1163 return;
1164 }
1165
1166 act_argc = argc;
1167 for(int i=act_argc; i>0; i--)
1168 args[i] = stack->Pop();
1169
1170 if(stack->Peek() == &cssMisc::Void) // get rid of arg stop..
1171 stack->Pop();
1172 }
1173 else {
1174 ///////////////////////////////////////////////////////////////////////////
1175 // has default args -- need to search for how many args we got
1176 int max_stack = stack->size - argc;
1177 for(stack_start = stack->size-1; stack_start >= max_stack; stack_start--) {
1178 if(stack->FastEl(stack_start) == &cssMisc::Void)
1179 break;
1180 }
1181 act_argc = ((int)stack->size - stack_start) - 1;
1182
1183 int diff = argc - act_argc;
1184 if(diff > arg_defs.size) { // more difference than we have defaults..
1185 cssMisc::Error(prog, "Incomplete argument list for:", (const char*)name,
1186 "should have at least:", StringtaString((int)(argc-arg_defs.size)), "got:",
1187 StringtaString((int)act_argc));
1188 act_argc = -1; // err msg indication
1189 return;
1190 }
1191
1192 for(int i=act_argc; i>0; i--)
1193 args[i] = stack->Pop();
1194
1195 if(stack->Peek() == &cssMisc::Void) // get rid of arg stop..
1196 stack->Pop();
1197
1198 if(diff != 0) {
1199 int df_st = arg_defs.size - diff;
1200 for(int i=df_st; i < arg_defs.size; i++) // fill in using default args..
1201 args[def_start + i] = arg_defs.FastEl(i);
1202 }
1203 act_argc = argc; // we always get the right number...
1204 }
1205}

1940void cssMbrScriptFun::Define(cssProg* prg, bool decl, const StringtaString& nm) {
1941 if(nm.nonempty())

1943 // code was previously defined, and needs to be overwritten
1944 if((fun->size > 0) || (decl && (fun->Autos(0)->size > 0))) {
1945 cssProg* old_fun = fun;

1947 cssProg::Ref(fun);

1951 cssProgSpace* old_top = fun->SetTop(prg->top); // propagate top
1952 fun->name = type_def->name + "::" + name;
1953
1954 // all the args are on the stack, with the return value at the end (not start)
1955 cssSpace* stack = prg->Stack();
1956 cssRef* retval_ptr = (cssRef*)stack->Pop(); // get this for later..
1957
1958 int stack_start; // where to start getting things off of the stack
1959 for(stack_start = stack->size-1; stack_start >= 0; stack_start--) {
1960 if(stack->FastEl(stack_start) == &cssMisc::Void)
1961 break;
1962 }

1964
1965 if(decl) {

1967 cssMisc::Warning(NULL__null, "Warning: redefining function arg count to:", StringtaString(actual_argc),
1968 "from:", StringtaString(argc));
1969 }
1970 }
1971
1972 if(fun->Autos(0)->size > 0) { // already defined
1973 if(actual_argc != argc) {
1974 cssMisc::Warning(NULL__null, "Function definition arg count (", StringtaString(actual_argc),
1975 ") does not match declaration:", StringtaString(argc));
1976 }
1977 // get the original ones..
1978 cssElPtr elp;
1979 elp.SetProgAuto(fun);
1980 int i;
1981 for(i=0; i<=argc; i++) {
1982 elp.dx = i; // index into auto's
1983 argv[argc-i] = elp; // set argv pointer..
1984 }
1985 // then check with current ones..
1986 if(actual_argc == argc) {
1987 for(i=argc; i>1; i--) { // don't write over 1st 2 (0 and 1)
1988 cssRef* tmp = (cssRef*)stack->Pop();
1989 cssEl* carv = argv[i].El();
1990 if(carv->GetType() != tmp->ptr.El()->GetType()) {

2009 cssRef* tmp = (cssRef*)stack->Pop();

2011 cssEl::Done(tmp);
2012 }
2013 if(stack->Peek() == &cssMisc::Void) // get rid of arg stop..
2014 stack->Pop();
2015
2016 // install a pointer to hold the "this" pointer
2017 cssPtr* ths_ptr = new cssPtr();
2018 ths_ptr->name = "this"; // its actually called this..

2020
2021 argv[0] = fun->AddAuto(retval_ptr->ptr.El()); // retval
2022 }
2023
2024 cssEl::Done(retval_ptr);
