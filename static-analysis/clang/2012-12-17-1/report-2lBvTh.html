23/* note: we handle the following line end types:
24 Windows cr lf
25 MacOS cr
26 Unix lf
27
28 NOTE: crlf is treated as a single character -- last_char will be the char
29 before the cr, not cr
30*/
31int MTA::Getc() {
32 char c = fh.get();
33 // strm_pos++;
34 strm_pos = strm_pos + streampos(1);
35 // now, detect for CRLF and skip to the lf
36 if(verbose > 4) // && (state != MTA::Find_Item))
37 cerr << "C!!: => " << line << " :\t" << c << "\n";
38 if (c == '\r') {
39 if (Peekc() == '\n')
40 c = fh.get();
41 strm_pos = strm_pos + streampos(1);
42 }
43 if ((c == '\n') || (c == '\r')) {
44 MTA::LastLn[col] = '\0'; // always terminate
45 if(verbose > 3) // && (state != MTA::Find_Item))
46 cerr << "I!!: => " << line << " :\t" << MTA::LastLn << "\n";
47 line++;
48 st_line_pos = strm_pos;
49 col = 0;
50 } else {
51 if(col >= 8191)
52 cout << "W!!: Warning, line length exceeded in line: " << line << "\n";
53 else {
54 MTA::LastLn[col++] = c;
55 MTA::LastLn[col] = '\0'; // always terminate
56 }
57 }
58 return c;
59}

103int MTA::readword(int c) {
104 LexBuf = "";
105
106 LexBuf += (char)c;
107 while ((c=Peekc()) != EOF(-1) && (isalnum(c) || (c == '_'))) { LexBuf += (char)c; Getc(); }
108 if(c == EOF(-1))
109 return EOF(-1);
110 if ((c == '\n') || (c == '\r'))
111 Getc();
112 return c;
113}

115int MTA::readfilename(int c) {
116 // if c is a quote, then read to end quote, else assume it is unquoted fname
117 // and read until next ws
118 // note that quoted fnames can contain embedded spaces
119 LexBuf = "";
120 if (c == '\"') { // since c was the opening ", we don't add it to lexbuf
121 // note: ws inside will be Very Bad, but we terminate regardless
122 while (((c=Getc()) != EOF(-1)) && (c != '\t') && (c != '\n') && (c != '\r')) {
123 if (c == '\"') { // found terminator, so skip it and break out
124 c = Getc();
125 break;
126 }
127 LexBuf += (char)c;
128 }
129 } else {
130 LexBuf += (char)c; // gotta include the first guy!
131 while (((c=Getc()) != EOF(-1)) && (c != ' ') && (c != '\t') && (c != '\n') && (c != '\r')) {
132 LexBuf += (char)c;
133 }
134 }
135 if(c == EOF(-1))
136 return EOF(-1);
137 return c;
138}
