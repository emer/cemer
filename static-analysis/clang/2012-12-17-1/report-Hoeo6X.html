189/*
190 Object on this list will only have refs if they are owned, ex. in a list.
191 If the owner or mgr nukes the obj before we get to it, it will have been
192 automatically removed from the delayed_close list. Otherwise, we will
193 "nuke with extreme prejudice".
194 Note that it might be possible for objects to get added to the list
195 while deleting something -- we won't do those this cycle.
196*/
197bool tabMisc::DoDelayedCloses() {
198 if (delayed_close.size == 0) return false;
199 // note: this is sleazy, but very efficient -- we just completely
200 // hijack the memory of the current list in this working copy
201 taBase_RefList items;
202 items.Hijack(delayed_close);
203 // we work fifo
204 while (items.size > 0) {
205 // note: active items can often have many refs, and CutLinks will typically
206 // resolve those (ex. Projection, which can have many cons)
207 // so if item is owned, we just defer to it, otherwise
208 taBase* it = items.FastEl(0);
209 items.RemoveIdx(0);
210 int refn = taBase::GetRefn(it);
211 if ((it->GetOwner() == NULL__null) && (refn != 1)) {
212 taMisc::DebugInfo("tabMisc::delayed_close: item had refn != 1, was=",
213 StringtaString(refn), "type=", it->GetTypeDef()->name, "name=",
214 it->GetName());
215 }
216 it->Close();
217 }
218 return true;
219}

279/*!
280 \class taBase ta_base.h
281 \brief The root class for most other classes.
282
283 \ingroup basic
284 \mainclass
285
286 \target notifications
287 \section1 Change Notifications
288
289 The taBase class provides a data update/change mechanism in conjunction
290 with the taDataLink class. Note that the term "attributes" is used to
291 refer to any member that controls the way the object behaves or looks;
292 "dynamic state" on the other hand, is the set of members that describe
293 the object's current parameters, ex. activation for a neuron. State
294 changes usually don't invoke the following mechanisms.
295
296 The update mechanism supports the following
297 types of changes:
298
299 UpdateAfterEdit -- a change that occurs after a user or script updates
300 an attribute of an object. A UAE signaled during a batch update (see below)
301 is generally ignored; batch updates therefore need to encompass the
302 changes made by an UAE call.
303
304 Batch data update -- this is a nestable update that enables any number
305 of nested data changes; the update notification is deferred until all
306 updates are complete. A data update implies the change to data members,
307 including nested data members, but in general does not include any
308 structural changes, such as adding or deleting child objects.
309
310 Batch structure update -- this is a more wide-ranging kind of update
311 that includes Batch Data update as a subset, but also comprises changes
312 to data members, such as additions or deletions in lists.
313
314 List/Group member insertion -- a container is notified when a new item
315 is added.
316
317 Deletion -- a link receives a notification when an object is about to
318 be destroyed.
319
320 The following situations are handled automatically, you don't need to
321 do anything in your code: List/Group member changes; object deletion.
322 The other cases require explicit handling in the code, as follows :
323
324 UAE -- call this->UpdateAfterEdit() after attribute changes that could
325 affect the UI or other objects. But *don't* call UAE when updating
326 state variables as part of simulations, etc. -- those have their own
327 display update mechanism.
328
329 Batch or Struct Data Update -- call the update in pairs, around the
330 code in which the changes are taking place (use StructUpdate if
331 the change is structural):
332 \code
333 this->DataUpdate(true);
334 \\ data changes go here
335 this->DataUpdate(false);
336 \endcode
337
338 \section2 Lifetime Management
339
340 taBase object lifetimes can be either Statically, or Dynamically
341 managed, and the visibility can either be Internal or External.
342 Static objects are created and destroyed by the compiler; Dynamic
343 objects are created/destroyed by the programmer (new/delete).
344 Internal objects are used solely by the managing owner, whereas
345 External objects can be accessed by an external client.
346 In practice, this gives rise to three lifetime/visibility scenarios:
347 Static External, Dynamic External, and Internal.
348
349 Internal objects need not be explicitly lifetime managed, and may
350 be created statically or dynamically. Typically, refn remains 0
351 for these objects, and no ref counting occurs for them.
352
353 External objects can be uniformly treated as if they obeyed ref
354 counting semantics. This enables external references to follow a
355 single paradigm. Of course the lifetime of static objects must be
356 respected, but the ref counting mechanism will issue a warning if
357 a "live" static object is deleted.
358
359 Ref Counting Mechanism -- Ref() causes refn++, UnRef() causes refn--;
360 if refn goes from 1 to 0, the object deletes. The transition from
361 1 to 0 also causes refn to be set to a -ve sentinel value. The debug
362 version of the program can detect double destruction, or ref count
363 operations after destruction, based on this sentinel.
364
365 Statically Managed -- refn is set to 1 after creation (see below);
366 Ref() causes refn++, UnRef() causes refn--; there
367 should be no unref that causes refn to go from 1 to 0, so refn s/b
368 1 on destruction.
369
370 The basic rule is that statically managed objects should always
371 be given an Own() operation, or a RefStatic() operation, after
372 creation, to insure the +1 refn is set. There is no corresponding
373 unrefing operation.
374
375 Most statically owned objects have an Own() operation performed on
376 them in the InitLinks routine. Own(obj&, owner*) is basically used
377 only when the obj is a static member of owner; therefore, this
378 version of Own automatically calls RefStatic(). It is ok for RefStatic
379 to get called multiple times during initialization of an object --
380 this enables new objects to get a RefStatic put into their owner's
381 Initialize. Own() can detect a change in owner, for objects that
382 provide the owner -- this lets it skip the RefStatic() operation.
383 The above semantics should cover all normal cases.
384*/

760taBase::IndexMode taBase::IndexModeDecode(const Variant& idx, int cont_dims) const {
761 IndexMode mode = IDX_UNK;
762 if(idx.isStringType()) {
763 mode = IDX_NAME;
764 }
765 else if(idx.isNumericStrict()) {
766 mode = IDX_IDX;
767 }
768 else if(idx.isMatrixType()) {
769 taMatrix* mat = idx.toMatrix();
770 if(TestError(!mat, "IndexModeDecode", "matrix index is NULL"))
771 return IDX_UNK;
772 if(mat->InheritsFrom(&TA_String_Matrix)) {
773 mode = IDX_NAMES;
774 }
775 else if(mat->InheritsFrom(&TA_byte_Matrix)) {
776 mode = IDX_MASK;
777 }
778 else if(mat->InheritsFrom(&TA_slice_Matrix)) {
779 mode = IDX_SLICE;
780 }
781 else if(mat->InheritsFrom(&TA_int_Matrix)) {
782 if(cont_dims == 1) {
783 if(mat->dims() == 1 && mat->dim(0) == 1) {
784 mode = IDX_COORD;
785 }
786 else if(mat->dims() == 1 || (mat->dims() == 2 && mat->dim(0) == 1)) {
787 mode = IDX_COORDS;
788 }
789 else if(mat->dims() == 2 && mat->dim(0) == 3 && mat->dim(1) == 1) {
790 mode = IDX_SLICE;
791 }
792 }
793 else {
794 if(mat->dims() == 1 && mat->dim(0) == cont_dims) { // dims must be exact match
795 mode = IDX_COORD;
796 }
797 else if(mat->dims() == 2 && mat->dim(0) == 3 &&
798 (mat->dim(1) >= 1 && mat->dim(1) <= cont_dims)) {
799 mode = IDX_SLICE;
800 }
801 else if(mat->dims() == 2 && mat->dim(0) == cont_dims) {
802 mode = IDX_COORDS;
803 }
804 if(mode == IDX_UNK) {
805 StringtaString matstr;
806 mat->Print(matstr);
807 TestError(true, "Index Decode",
808 "index access failed, most likely because the number of dimensions provided did not match the dimensionality of the container:", StringtaString(cont_dims), " index provided was:", matstr);
809 }
810 }
811 }
812 else {
813 mode = IDX_MISC; // typically a Variant_Matrix
814 }
815 }
816 return mode;
817}

1198taBase* taBase::FindFromPath(const StringtaString& path, MemberDef*& ret_md, int start) const {
1199 if(((int)path.length() <= start) || (path == ".")) {
1200 ret_md = NULL__null;
1201 return (taBase*)this;
1202 }
1203 if((path == "Null") || (path == "NULL")) {
1204 ret_md = NULL__null;
1205 return NULL__null;
1206 }
1207
1208 taBase* rval = NULL__null;
1209 bool ptrflag = false;
1210 int length = path.length();
1211
1212 while(start < length) {
1213 if(path[start] == '*') { // path is a pointer
1214 start += 2;
1215 ptrflag = true;
1216 }
1217 if(path[start] == '.') { // must be root, so search on next stuff
1218 start++;
1219 continue;
1220 }
1221
1222 int delim_pos = taBase::GetNextPathDelimPos(path, start);
1223 StringtaString el_path = path(start,delim_pos-start); // element is between start and delim
1224 int next_pos = delim_pos+1;
1225 if((delim_pos < length) && (path[delim_pos] == '['))
1226 next_pos--;
1227
1228 MemberDef* md;
1229 void* tmp_ptr = FindMembeR(el_path, md);
1230 if(tmp_ptr && (!md || md->type->InheritsFrom(TA_taBase))) { // null md = taBase
1231 taBase* mbr = (taBase*)tmp_ptr;
1232 if(delim_pos < length) { // there's more to be done..
1233 rval = mbr->FindFromPath(path, ret_md, next_pos); // start from after delim
1234 }
1235 else {
1236 rval = mbr; // that's all folks..
1237 ret_md = md;
1238 }
1239 }
1240 else if((el_path == "root") && (delim_pos < length)) {
1241 start = next_pos; // skip this element since it must be us
1242 continue;
1243 }
1244 if((ptrflag) && (rval != NULL__null))
1245 return *((taBase* *)rval);
1246 return rval;
1247 }
1248 return NULL__null;
1249}

1293 int point_idx = taMisc::find_not_in_quotes(path, '.', start+1); // skip any possible starting delim
1294 int brack_idx = taMisc::find_not_in_quotes(path, '[', start+1);
1295
1296 // if there is a period but not a bracket, or the period is before the bracket
1297 if(((brack_idx < start) && (point_idx >= start)) ||
1298 ((point_idx < brack_idx ) && (point_idx >= start)))
1299 {
1300 return point_idx;
1301 }
1302 else if(brack_idx >= start) { // else try the bracket
1303 return brack_idx;
1304 }
1305 return path.length(); // delimiter is end of string (its all element)
1306}

1500int taBase::Load_cvt(taFiler*& flr) {
1501 int c = taMisc::read_till_eol(*flr->istrm);
1502 if(c == EOF(-1)) return false; // bail
1503 if(!taMisc::LexBuf.contains("// ta_Dump File v1.0")) {
1504 flr->Close();

1513 }
1514 int typ_id = taMisc::find_strings(*flr->istrm, key_strs);
1515 flr->Close(); // file is done with anyway
1516 flr->open_read(); // read again in any case

1518 "Old format file could not be identified; not converting!")) return false;

1520 taFiler* cvt_flr = taFiler::New(flr->filetype, flr->defExt());
1521 taRefN::Ref(cvt_flr);

1525 if(!flr->defExt().empty()) {
1526 if (cvt_fname.contains(flr->defExt())) {
1527 cvt_fname = cvt_fname.before(flr->defExt());
1528 }
1529 cvt_fname += cvt_tag + flr->defExt();
1530 }
1531 else {
1532 StringtaString ex;
1533 if(cvt_fname.contains('.')) {
1534 ex = cvt_fname.after('.',-1);
1535 cvt_fname = cvt_fname.before('.',-1);

1537 cvt_fname += cvt_tag + "." + ex;
1538 }
1539 cvt_flr->SetFileName(cvt_fname);

1543 cvt_flr->Close();
1544 taRefN::unRefDone(flr); // get rid of orig filer
1545 flr = cvt_flr; // use new one
1546 flr->open_read(); // read the converted file
1547 int chs = taMisc::Choice("Note: converting old file of type: " + cvt->proj_type_base +
1548 ".\n Created intermediate cvt file as: " + cvt_fname +
1549 ".\n Many error messages are likely (and should be ignored)," +
1550 "\n and you will probably need to convert object using convert button",
1551 "Continue", "Cancel");

1554 else return false;

5789 }
5790 if(!IndexModeValidate(idx, mode, 1))
5791 return _nilVariantVariant();
5792 switch(mode) {
5793 case IDX_IDX: {
5794 return SafeElAsVar(idx.toInt());

5796 }
5797 case IDX_NAME:
5798 case IDX_MISC:
5799 case IDX_NAMES: {

5815 int el_idx = mitm.toInt();
5816 if(el_idx < 0) el_idx += size;
5817 if(el_idx < 0 || el_idx >= size) continue; // skip out of range at this point
5818 imat->Add(el_idx);
5819 }
5820 taArray_base* nwvw = NewElView(imat, IDX_COORDS);
5821 return (Variant)nwvw;
5822 break;
5823 }
5824 case IDX_SLICE: {
5825 int_Matrix* cmat = dynamic_cast<int_Matrix*>(idx.toMatrix());

5827 int end = cmat->FastEl_Flat(1);

5841 }
