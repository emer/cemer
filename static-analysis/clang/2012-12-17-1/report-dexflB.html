515static cssEl* cssElCFun_member_fun_stub(int, cssEl* arg[]) {
516 cssProg* cp = arg[0]->prog;
517 cssEl* mbfun;
518 if(arg[2]->GetType() == cssEl::T_String)
519 mbfun = (arg[1])->GetMethodFmName(arg[2]->GetStr());
520 else
521 mbfun = (arg[1])->GetMethodFmNo((int)*arg[2]);
522 if((mbfun->GetType() != cssEl::T_MbrCFun) && (mbfun->GetType() != cssEl::T_MbrScriptFun)) {
523 if(mbfun != &cssMisc::Void)
524 cssEl::Done((cssEl*)mbfun);
525 cssMisc::Error(cp, "Member is not a function", mbfun->name);
526 return &cssMisc::Void;
527 }
528 // find the member_call, pointed to by the jump statement just after this guy.
529 cssInst* mbr_jmp = cp->insts[cp->PC()];
530 if(!mbr_jmp->IsJump()) {
531 cssMisc::Error(cp, "Internal error: member_fun didn't find jump stmt to find member_call");
532 return &cssMisc::Void;
533 }
534 css_progdx mbr_call_dx = mbr_jmp->GetJump();
535 if(cp->size <= mbr_call_dx) {
536 cssMisc::Error(cp, "Internal error: member_fun didn't find member_call instruction");
537 return &cssMisc::Void;
538 }
539
540 // code the new instruction
541 cp->ReplaceCode(mbr_call_dx, mbfun);
542
543 cp->Frame()->pc++; // skip over jump!
544
545 // now deal with args
546 cp->Stack()->Push(&cssMisc::Void); // push the arg stop
547 cssEl* ths = arg[1]->GetActualObj(); // get past references and pointers
548 if(mbfun->GetType() == cssEl::T_MbrScriptFun) { // push the 'this' pointer
549 if(ths->GetType() == cssEl::T_Class)
550 cp->Stack()->Push(ths);
551 else {
552 if(cp->CurThis() == NULL__null) {
553 cssMisc::Error(cp, "No 'this' pointer for member function call");
554 return &cssMisc::Void;
555 }
556 cp->Stack()->Push(cp->CurThis());
557 }
558 }
559 // push the 'this' pointer for builtin functions
560 else if(mbfun->GetType() == cssEl::T_MbrCFun) {
561 // just always push on stack because otherwise the this gets nuked!!!
562// if(ths->GetType() == cssEl::T_Class)
563 cp->Stack()->Push(ths);
564 }
565 if(cp->top->debug >= 2) {
566 taMisc::Info("\nCalling member function:", mbfun->name);
567 }
568 return &cssMisc::Void;
569}
