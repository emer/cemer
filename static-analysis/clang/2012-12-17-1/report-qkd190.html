69static int ctrl_char(int c) {
70 int rval = ' ';
71 switch (c) {
72 case 'n':
73 rval = '\n'; break;
74 case 't':
75 rval = '\t'; break;
76 case 'v':
77 rval = '\v'; break;
78 case 'b':
79 rval = '\b'; break;
80 case 'r':
81 rval= '\r'; break;
82 case 'f':
83 rval = '\f'; break;
84 case '\\':
85 rval = '\\'; break;
86 case '?':
87 rval = '\?'; break;
88 case '\'':
89 rval = '\''; break;
90 case '\"':
91 rval = '\"'; break; // "
92 case '0':
93 rval = '\0'; break;
94 }
95 return rval;
96}
