65void HCBar::paintEvent(QPaintEvent* ev) {
66 if (!scale) return; // some kind of zombie happening...
67 QRect r = rect();
68 int w_tot = r.width();
69 if (w_tot == 0) return; // prob should never happen
70
71 int b = blocks(); // cache
72 int x = r.x(); // s/b 0
73 int h = r.height(); // always exact
74 int y = r.y(); // always exact
75 QPainter paint(this);
76 if (w_tot <= b ) { // less or same # strips as blocks -- subsample
77 for (int j = 0; j < w_tot ; ++j) { // one strip per pixel
78 int i = (j * (b - 1)) / w_tot ;
79 bool ok;
80 const iColor col = scale->GetColor(i, &ok);
81 if (!ok) break; // shouldn't happen
82 paint.fillRect(x, y, 1, h, (QColor)col); // note: QColor converted to QBrush by Qt
83 x += 1;
84 }
85 } else { // more space than blocks -- oversample
86 float w = ((float)w_tot) / b; // ideal exact value per strip
87 for (int i = 0; i < b; ++i) { // one strip per color value
88 bool ok;
89 const iColor col = scale->GetColor(i, &ok);
90 if (!ok) break; // shouldn't happen
91 int wi = (int)(((i + 1) * w) - (float)x);
92
93 paint.fillRect(x, y, wi, h, (QColor)col); // note: QColor converted to QBrush by Qt
94 x += wi;
95 }
96 }
97}
