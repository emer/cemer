229void SNrThalLayerSpec::Compute_GateActs(LeabraLayer* lay, LeabraNetwork* net) {
230 if(net->ct_cycle < snrthal.gate_cycle)
231 return; // nothing to do before gating starts
232
233 Layer::AccessMode acc_md = Layer::ACC_GP;
234 int nunits = lay->UnitAccess_NUnits(acc_md); // this should be just 1 -- here for generality but some of the logic doesn't really go through for n >= 2 at this point..
235
236 int n_fired_trial = 0;
237 int n_fired_now = 0;
238
239 for(int mg=0; mg<lay->gp_geom.n; mg++) {
240 PBWMUnGpData* gpd = (PBWMUnGpData*)lay->ungp_data.FastEl(mg);
241 LeabraUnit* u = (LeabraUnit*)lay->UnitAccess(acc_md, 0, mg); // assuming one unit
242 if(u->lesioned()) continue;
243
244 if(net->ct_cycle == snrthal.gate_cycle) {
245 if(u->act_eq >= snrthal.go_thr) {
246 n_fired_trial++;
247 n_fired_now++;
248 gpd->go_fired_now = true;
249 gpd->go_fired_trial = true;
250 gpd->go_cycle = net->ct_cycle;
251 gpd->prv_mnt_count = gpd->mnt_count;
252 gpd->mnt_count = 0; // reset
253 u->act_m2 = u->act_eq; // gating act
254 }
255 }
256 else {
257 gpd->go_fired_now = false; // turn it off after one cycle
258 u->act = u->act_eq = u->act_nd = u->act_m2; // activity is always gating signal
259 }
260 }
261
262 // no stripe has fired yet -- force max go firing
263 if(net->ct_cycle == snrthal.gate_cycle && snrthal.force && n_fired_trial == 0) {
264 int go_idx = lay->netin.max_i;
265 if(go_idx < 0) { // this really shouldn't happen, but if it does..
266 go_idx = Random::IntZeroN(lay->units.leaves);
267 }
268 int gp_idx = go_idx / nunits;
269 LeabraUnit* u = (LeabraUnit*)lay->UnitAccess(acc_md, 0, gp_idx); // assuming one unit
270 PBWMUnGpData* gpd = (PBWMUnGpData*)lay->ungp_data.FastEl(gp_idx);
271 n_fired_trial++;
272 n_fired_now++;
273 gpd->go_fired_now = true;
274 gpd->go_fired_trial = true;
275 gpd->go_forced = true; // yes this is forced
276 gpd->go_cycle = net->ct_cycle;
277 gpd->prv_mnt_count = gpd->mnt_count;
278 gpd->mnt_count = 0; // reset
279 u->act_m2 = snrthal.go_thr; // must be at least at threshold -- otherwise won't gate for pbwm!
280 }
281
282 if(net->ct_cycle == snrthal.gate_cycle) {
283 Compute_GateStats(lay, net); // update overall stats at this point
284 lay->SetUserData("n_fired_trial", n_fired_trial);
285 lay->SetUserData("n_fired_now", n_fired_now);
286 }

869 gpd->pvr_inhib = pvr_i;
870
871 float mult_fact = 1.0f;
872 if(nogo_i > 0.0f) mult_fact *= (1.0f - nogo_i);
873 if(noact_i > 0.0f) mult_fact *= (1.0f - noact_i);
874 if(refract_i > 0.0f) mult_fact *= (1.0f - refract_i);
875 if(pvr_i > 0.0f) mult_fact *= (1.0f - pvr_i);

878 for(int j=0;j<nunits;j++) {

880 if(u->lesioned()) continue;
881 u->net *= mult_fact;

884 }
885 }

1062void PFCsUnitSpec::Compute_ActFmVm_rate(LeabraUnit* u, LeabraNetwork* net) {
1063 LeabraLayer* rlay = u->own_lay();
1064 if(rlay->lesioned()) return;
1065 if(!rlay->GetLayerSpec()->InheritsFrom(&TA_PFCLayerSpec)) return;
1066 PFCLayerSpec* ls = (PFCLayerSpec*) rlay->GetLayerSpec();
1067 int rgpidx = u->UnitGpIdx();
1068 PBWMUnGpData* gpd = (PBWMUnGpData*)rlay->ungp_data.FastEl(rgpidx);
1069
1070 float new_act;
1071 if(act.gelin) {
1072 if(act.old_gelin) {
1073 float g_e_val = u->net;
1074 float vm_eq = act.vm_mod_max * (Compute_EqVm(u) - v_m_init.mean); // relative to starting!
1075 if(vm_eq > 0.0f) {
1076 float vmrat = (u->v_m - v_m_init.mean) / vm_eq;
1077 if(vmrat > 1.0f) vmrat = 1.0f;
1078 else if(vmrat < 0.0f) vmrat = 0.0f;
1079 g_e_val *= vmrat;
1080 }
1081 float g_e_thr = Compute_EThresh(u);
1082 new_act = Compute_ActValFmVmVal_rate(g_e_val - g_e_thr);
1083 // NEW CODE: maintaining
1084 if(gpd->mnt_count > 0) {
1085 u->misc_1 = new_act;
1086 new_act = ls->gate.maint_pct * u->maint_h + ls->gate.maint_pct_c * new_act;
1087 }
1088 else {
1089 u->misc_1 = 0.0f;
1090 }
1091
1092 }
1093 else { // new gelin
1094 if(u->v_m <= act.thr) {
1095 new_act = Compute_ActValFmVmVal_rate(u->v_m - act.thr);
1096 }
1097 else {
1098 float g_e_thr = Compute_EThresh(u);
1099 new_act = Compute_ActValFmVmVal_rate(u->net - g_e_thr);
1100 }
1101 // NEW CODE: maintaining
1102 if(gpd->mnt_count > 0) {
1103 u->misc_1 = new_act;
1104 new_act = ls->gate.maint_pct * u->maint_h + ls->gate.maint_pct_c * new_act;
1105 }
1106 else {
1107 u->misc_1 = 0.0f;
1108 }
1109
1110 if(net->cycle < dt.vm_eq_cyc) {
1111 new_act = u->act_nd + dt.vm_eq_dt * (new_act - u->act_nd); // eq dt
1112 }
1113 else {
1114 new_act = u->act_nd + dt.vm * (new_act - u->act_nd); // time integral with dt.vm -- use nd to avoid synd problems
1115 }
1116 }
1117 }
1118 else {
1119 new_act = Compute_ActValFmVmVal_rate(u->v_m - act.thr);
1120 // NEW CODE: maintaining
1121 if(gpd->mnt_count > 0) {
1122 u->misc_1 = new_act;
1123 new_act = ls->gate.maint_pct * u->maint_h + ls->gate.maint_pct_c * new_act;
1124 }
1125 else {
1126 u->misc_1 = 0.0f;
1127 }
1128 }
1129 if(depress.on) { // synaptic depression
1130 u->act_nd = act_range.Clip(new_act); // nd is non-discounted activation!!! solves tons of probs
1131 new_act *= MIN(u->spk_amp, 1.0f)(((u->spk_amp) < (1.0f)) ? (u->spk_amp) : (1.0f));
1132 if((net->ct_cycle+1) % depress.interval == 0) {
