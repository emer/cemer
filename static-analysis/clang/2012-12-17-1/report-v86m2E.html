41void TypeSpace_Generate_AddUndef(TypeSpace* ths, TypeSpace* trg, TypeSpace* dst) {
42 int i;
43 for(i=0; i< ths->size; i++) {
44 TypeDef* td = ths->FastEl(i);
45 if((td->owner == NULL__null) || (td->owner == trg) || (td->pre_parsed)
46 || (td->owner->owner != NULL__null))
47 continue;
48
49 TypeDef* utd = dst->AddUniqNameOld(td);
50 utd->internal = true; // make internal
51 if(utd == td) { // add to dst
52 if((!td->DerivesFormal(TA_class) && !td->DerivesFormal(TA_enum))
53 || (td->ptr > 0)) // add this stuff when unique
54 TypeSpace_Generate_AddUndef(&(td->parents), trg, dst); // non-class parents are added..
55 }
56 }
57}

122bool TypeSpace_Sort_Order(TypeSpace* ths) {
123 bool move_occurred = false;
124 int i;
125 for(i=0; i< ths->size; i++) {
126 TypeDef* td = ths->FastEl(i);
127 int j;
128 for(j=0; j<td->parents.size; j++) {
129 TypeDef* par_td = td->parents.FastEl(j);
130 if((td->idx < par_td->idx) && (td->owner == ths) && (par_td->owner == ths)) {
131 if(mta->verbose > 0)
132 cerr << "M!!: Switching order of: " << td->name << " fm: " << td->idx << " to: "
133 << par_td->idx+1 << "\n";
134 // child comes before parent..
135 ths->MoveIdx(td->idx, par_td->idx+1); // move after parent
136 move_occurred = true;
137 }
138 }
139 }
140 return move_occurred;

656 int args_idx = j + stub_arg_off;
657
658 TypeDef* class_typ = argt->GetClassType();
659
660 if(argt->ref) {
661 if(!argt->InheritsFrom(TA_const)) { // non-const reference arg!

663 }
664 else {
665 TypeDef* nrt = argt->GetNonRefType();
666 if(nrt == NULL__null) {
667 taMisc::Error("Null NonRefType in GenArgCast()", argt->name); return;

674 return;
675 }
676 }
677 else if(argt->DerivesFrom(TA_TypeDef)) { // we convert these now
678 strm << "(TypeDef";
679 strm << argt->GetPtrString() << ")";
680 strm << "*arg[" << args_idx << "]";
681 }
682 else if(argt->DerivesFrom(TA_MemberDef)) { // we convert these now
683 strm << "(MemberDef";
684 strm << argt->GetPtrString() << ")";
685 strm << "*arg[" << args_idx << "]";

690 strm << "*arg[" << args_idx << "]";
691 }

693 if(argt->ptr == 0)
694 strm << "arg[" << args_idx << "]->GetStr()";
695 else
696 strm << "(String" << argt->GetPtrString() << ")*arg[" << args_idx << "]";
697 }

699 if(argt->ptr == 0)
700 strm << "arg[" << args_idx << "]->GetVar()";
701 else
702 strm << "(Variant" << argt->GetPtrString() << ")*arg[" << args_idx << "]";
703 }
704 else if(argt->DerivesFromName("ios")) { // cssEl's can cast these directly
705 if(argt->ptr == 0)
706 strm << "*(" << argt->Get_C_Name() << "*)" << "*arg[" << args_idx << "]";
707 else
708 strm << "(" << argt->Get_C_Name() << ")" << "*arg[" << args_idx << "]";

715 strm << "(" << argt->Get_C_Name() << ")" << "*arg[" << args_idx << "]";

718 strm << "(" << argt->Get_C_Name() << ")"; // always cast the args
719 strm << "(int)*arg[" << args_idx << "]"; // use int conversion
720 }
721 else if(class_typ) { // is some kind of class object -- just cast away!
722 strm << "(" << argt->Get_C_Name() << ")arg[" << args_idx << "]"
723 << "->GetVoidPtrOfType(&TA_" << class_typ->name << ")";
724 }

726 // this is extremely dangerous, but hopefully will just fail
727 strm << "(" << argt->Get_C_Name() << ")(void";
728 strm << argt->GetPtrString() << ")";
729 strm << "*arg[" << args_idx << "]";
730 }
731 else {
732 strm << "(" << argt->Get_C_Name() << ")"; // always cast the args
733 strm << "*arg[" << args_idx << "]";
734 }

1378void TypeDef_Generate_AddParents(TypeDef* ths, char* typ_ref, ostream& strm) {
1379 TypeDef_Generate_AddOtherParents(ths, typ_ref, strm);
1380
1381// (u)intptr_t requires test at runtime:
1382 if (ths == &TA_intptr_t) {
1383 strm << " if (sizeof(intptr_t) == sizeof(int)) " << typ_ref
1384 << "AddParents(&TA_int);\n else " << typ_ref << "AddParents(&TA_int64_t);\n";
1385 return;
1386 } else if (ths == &TA_uintptr_t) {
1387 strm << " if (sizeof(uintptr_t) == sizeof(unsigned int)) " << typ_ref
1388 << "AddParents(&TA_unsigned_int);\n else " << typ_ref << "AddParents(&TA_uint64_t);\n";
1389 return;
1390 } else
1391// long types just get parented and thus aliased to the correct size
1392 if (ths == &TA_long) {
1393 strm << " if (sizeof(long) == sizeof(int)) " << typ_ref
1394 << "AddParents(&TA_int);\n else " << typ_ref << "AddParents(&TA_int64_t);\n";
1395 return;
1396 } else if (ths == &TA_unsigned_long) {
1397 strm << " if (sizeof(unsigned long) == sizeof(unsigned int)) " << typ_ref
1398 << "AddParents(&TA_unsigned_int);\n else " << typ_ref << "AddParents(&TA_uint64_t);\n";
1399 return;
1400 }
1401 if(ths->parents.size == 0)
1402 return;
1403
1404 int cnt=0;
1405 int i;
1406 // see if there are any parents, and also check for too many for us to handle!
1407 for (i=0; i < ths->parents.size; i++) {
1408 TypeDef* ptd = ths->parents.FastEl(i);
1409/*NOTE: following is very gruesome...
1410 in short, we don't want to add parent TypeDef's that don't actually exist,
1411 BUT, we can have them from earlier files, in which they will be in
1412 the spc_pre_parse list -- so we need to generate if in there
1413 (earlier version 3.x handled this wrong -- the use cases are fairly
1414 infrequent, but critical, typical example is where a base class is defined
1415 in an early module, and things like subclasses or pointers to that class
1416 only defined in a later module)
1417*/
1418 if ((ptd->owner != ths->owner) && !ptd->pre_parsed &&
1419 (mta->spc_builtin.FindName(ptd->name) == NULL__null) &&
1420 (mta->spc_pre_parse.FindName(ptd->name) == NULL__null)) {
1421 if (mta->verbose > 0) {
1422 cerr << "M!!: maketa: ("
1423 << (ptd->owner != ths->owner) << ", "
1424 << !ptd->pre_parsed << ", "
1425 << (mta->spc_builtin.FindName(ptd->name) == NULL__null)
1426 << ") skipping parent '" << ptd->name << "' for: " << ths->name << endl;
1427 if (ptd->owner) cerr << " ptd space: " << ptd->owner->name << endl;
1428 if (ths->owner) cerr << " ths space: " << ths->owner->name << endl;
1429 }
1430 continue; // add parents only if on same list. (except if pre-parsed or builtin)
1431 }
1432 cnt++;
1433 }
1434 if (cnt == 0)
1435 return;
1436 if (cnt > PAR_ARG_COUNT6) {
1437 taMisc::Error("AddParents(): parents.size > ", StringtaString(PAR_ARG_COUNT6),
1438 ", increase number of args to AddParents()", "type name:", ths->name);
1439 return;
1440 }
1441
1442 strm << " " << typ_ref;
1443 if(ths->InheritsFormal(TA_class) && !ths->internal) strm << "AddClassPar(";
1444 else strm << "AddParents(";
1445 StringtaString ths_cnm = ths->Get_C_Name();
1446 cnt = 0;
1447 for (i=0; i < ths->parents.size; ++i) {
1448 TypeDef* ptd = ths->parents.FastEl(i);
1449 if ((ptd->owner != ths->owner) && !ptd->pre_parsed &&
1450 (mta->spc_builtin.FindName(ptd->name) == NULL__null) &&
1451 (mta->spc_pre_parse.FindName(ptd->name) == NULL__null)) {
1452 continue; // add parents only if on same list. (except if pre-parsed)
1453 }
1454 if (cnt > 0)
1455 strm << ", ";
1456 if((ptd->owner != NULL__null) && (ptd->owner->owner != NULL__null))
1457 strm << "TA_" << ptd->owner->owner->name << ".sub_types.FindName(\"" << ptd->name << "\")";
1458 else
1459 strm << "&TA_" << ptd->name;
1460 if(ths->InheritsFormal(TA_class) && !ths->internal) {
1461 if((ths->parents.size>1) && !ths->InheritsFormal(TA_template) && !ptd->InheritsFormal(TA_template)) {
1462 if((mta->gen_instances || (ths->HasOption("INSTANCE")))
1463 && !(ths->HasOption("NO_INSTANCE")))
1464 {
1465 strm << ",(int)((unsigned long)((" << ptd->Get_C_Name() << "*)"
1466 << "TAI_" << ths->name << ")-(unsigned long)TAI_" << ths->name
1467 << ")";
1468 }
1469 else {
1470 strm << ",0";
1471 if((ths->parents.size > 1) && (i==0) && !ths->InheritsFromName("ios") &&
1472 !ths->HasOption("NO_MEMBERS"))
1473 taMisc::Error("warning: type:",ths->name,"has mult inherit but no instance",
1474 "-parent offset cannot be computed!");
1475 }
1476 }
1477 else
1478 strm << ",0";
1479 }
1480 ++cnt;
1481 }
1482 strm << ");\n";
1483}
