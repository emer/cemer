111// copy the source to the destination, returning true if successful
112bool copy_file(const char* src, const char* dst) {
113 bool rval = false;
114 fstream fsrc, fdst;
115 fsrc.open(src, ios::in | ios::binary);
116 if (!fsrc.is_open()) return false;
117 fdst.open(dst, ios::out | ios::binary);
118 if (!fdst.is_open()) goto exit1;
119 char c1;
120 // from stroustrup
121 while (fsrc.get(c1)) fdst.put(c1);
122 rval = true;
123//exit2:
124 fdst.close();
125exit1:
126 fsrc.close();
127 return rval;
128}

130// returns 'true' if both files exist and are the same
131bool files_same(const char* fname1, const char* fname2) {
132 bool rval = false;
133 fstream in1, in2;
134
135 in1.open(fname1, ios::in | ios::binary);
136 if (!in1.is_open()) goto exit2;
137 in2.open(fname2, ios::in | ios::binary);
138 if (!in2.is_open()) goto exit1;
139 char c1; char c2;
140 bool g1; bool g2;
141 while (true) {
142 g1 = 0 != in1.get(c1);
143 g2 = 0 != in2.get(c2);
144 if (!g1 && !g2) break; // same size, done
145 if (!(g1 && g2)) goto exit; // different sizes
146 if (c1 != c2) goto exit; // different content
147 }
148 rval = true;
149
150exit:
151 in2.close();
152exit1:
153 in1.close();
154exit2:
155 return rval;
156}

460void MTA::SetDesc(const char* comnt, StringtaString& desc, String_PArray& inh_opts,
461 String_PArray& opts, String_PArray& lists) {
462 StringtaString tmp = comnt;
463 tmp.gsub("\"", "'"); // don't let any quotes get through
464 StringtaString ud;
465 while(tmp.contains('#')) {
466 desc += tmp.before('#');
467 tmp = tmp.after('#');
468 if(tmp.contains(' '))
469 ud = tmp.before(' ');
470 else
471 ud = tmp;
472 tmp = tmp.after(' ');
473 if(ud(0,5) == "LIST_") {
474 ud = ud.after("LIST_");
475 lists.AddUnique(ud);
476 }
477 else {
478 if(ud(0,1) == '#') {
479 ud = ud.after('#');
480 inh_opts.AddUnique(ud);
481 }
482 opts.AddUnique(ud);
483 }
484 }
485 desc += tmp;
486}

488TypeDef* MTA::FindName(const char* nm, int& lex_token) {
489 TypeDef* itm;
490
491 lex_token = TYPE269; // this is the default
492
493 if((state == Parse_inclass) && (cur_class != NULL__null)) {
494 if(cur_class->name == nm) {
495 lex_token = THISNAME283;
496 return cur_class;
497 }
498 if((itm = cur_class->sub_types.FindName(nm)) != NULL__null)
499 return itm;
500 if((itm = cur_class->templ_pars.FindName(nm)) != NULL__null)
501 return itm;
502 }
503
504 if(((itm = spc_keywords.FindName(nm)) != NULL__null) && (itm->idx != TA_TYPEDEF268)) {
505 lex_token = itm->idx;
506 return itm;
507 }
508
509 TypeDef *rval = NULL__null;
510 if((itm = spc->FindName(nm)) != NULL__null)
511 rval = itm;
512 else if((spc != &spc_other) && (itm = spc_other.FindName(nm)) != NULL__null)
513 rval = itm;
514 else if((itm = spc_ignore.FindName(nm)) != NULL__null)
515 rval = itm;
516
517 if(rval != NULL__null) {
518 if(rval == &TA_const)
519 lex_token = CONST266;
520 }
521 return rval;
522}
