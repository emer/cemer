564 else
565 recv_no = (int) ((p_con * ((float)prjn->from->units.leaves)) + .5f);
566 if(recv_no <= 0) recv_no = 1;
567
568 // sending number is even distribution across senders plus some imbalance factor
569 float send_no_flt = (float)(prjn->layer->units.leaves * recv_no) / (float)prjn->from->units.leaves;
570 // add SEM as corrective factor
571 float send_sem = send_no_flt / sqrtf(send_no_flt);
572 int send_no = (int)(send_no_flt + 2.0f * send_sem + 5.0f);

574
575 // pre-allocate connections!
576 prjn->layer->RecvConsPreAlloc(recv_no, prjn);
577 prjn->from->SendConsPreAlloc(send_no, prjn);
578
579 if((prjn->from.ptr() == prjn->layer) && sym_self) {
580 Layer* lay = prjn->layer;
581 // trick is to divide cons in half, choose recv, send at random
582 // for 1/2 cons, then go through all units and make the symmetric cons..
583 TestWarning(p_con > .95f, "Connect_impl", "there is usually less than complete connectivity for high values of p_con (>.95) in symmetric, self-connected layers using permute!");

585 int first;
586 if(!self_con)
587 first = (int) (.5f * p_con * (float)(prjn->from->units.leaves-1));
588 else
589 first = (int) (.5f * p_con * (float)prjn->from->units.leaves);
590 if(first <= 0) first = 1;

673float PolarRndPrjnSpec::UnitDist(UnitDistType typ, Projection* prjn,
674 const TwoDCoord& ru, const TwoDCoord& su)
675{
676 FloatTwoDCoord half(.5f);
677 PosTwoDCoord ru_geom = prjn->layer->flat_geom;
678 PosTwoDCoord su_geom = prjn->from->flat_geom;
679 switch(typ) {
680 case XY_DIST:
681 return ru.Dist(su);
682 case XY_DIST_CENTER: {
683 FloatTwoDCoord rctr = ru_geom; rctr *= half;
684 FloatTwoDCoord sctr = su_geom; sctr *= half;
685 TwoDCoord ruc = ru - (TwoDCoord)rctr;
686 TwoDCoord suc = su - (TwoDCoord)sctr;
687 return ruc.Dist(suc);
688 }
689 case XY_DIST_NORM: {
690 FloatTwoDCoord ruc = ru; ruc /= (FloatTwoDCoord)ru_geom;
691 FloatTwoDCoord suc = su; suc /= (FloatTwoDCoord)su_geom;
692 return ruc.Dist(suc);
693 }
694 case XY_DIST_CENTER_NORM: {
695 FloatTwoDCoord rctr = ru_geom; rctr *= half;
696 FloatTwoDCoord sctr = su_geom; sctr *= half;
697 FloatTwoDCoord ruc = ((FloatTwoDCoord)ru - rctr) / rctr;
698 FloatTwoDCoord suc = ((FloatTwoDCoord)su - sctr) / sctr;
699 return ruc.Dist(suc);
700 }
701 }
702 return 0.0f;
703}

705Unit* PolarRndPrjnSpec::GetUnitFmOff(UnitDistType typ, bool wrap, Projection* prjn,
706 const TwoDCoord& ru, const FloatTwoDCoord& su_off)
707{
708 FloatTwoDCoord half(.5f);
709 PosTwoDCoord ru_geom = prjn->layer->flat_geom;
710 PosTwoDCoord su_geom = prjn->from->flat_geom;
711 TwoDCoord suc; // actual su coordinates
712 switch(typ) {
713 case XY_DIST: {
714 suc = su_off;
715 suc += ru;
716 break;
717 }
718 case XY_DIST_CENTER: { // do everything relative to center
719 FloatTwoDCoord rctr = ru_geom; rctr *= half;
720 FloatTwoDCoord sctr = su_geom; sctr *= half;
721 TwoDCoord ruc = ru - (TwoDCoord)rctr;
722 suc = su_off;
723 suc += ruc; // add the centerized coordinates
724 suc += (TwoDCoord)sctr; // then add the sending center back into it..
725 break;
726 }
727 case XY_DIST_NORM: {
728 FloatTwoDCoord ruc = ru; ruc /= (FloatTwoDCoord)ru_geom;
729 FloatTwoDCoord suf = su_off + ruc; // su_off is in normalized coords, so normalize ru
730 suf *= (FloatTwoDCoord)su_geom;
731 suc = suf;
732 break;
733 }
734 case XY_DIST_CENTER_NORM: {
735 FloatTwoDCoord rctr = ru_geom; rctr *= half;
736 FloatTwoDCoord sctr = su_geom; sctr *= half;
737 FloatTwoDCoord ruc = ((FloatTwoDCoord)ru - rctr) / rctr;
738 FloatTwoDCoord suf = su_off + ruc;
739 suf *= sctr; suf += sctr;
740 suc = suf;
741 break;
742 }
743 }
744 if(suc.WrapClip(wrap, su_geom) && !wrap)
745 return NULL__null;
746
747 Unit* su_u = (Unit*)prjn->from->UnitAtCoord(suc);
748 return su_u;
749}

752float PolarRndPrjnSpec::GetDistProb(Projection* prjn, Unit* ru, Unit* su) {
753 if(rnd_dist.type == Random::UNIFORM)
754 return p_con;
755 float prob = p_con * rnd_dist.Density(UnitDist(dist_type, prjn, ru->pos, su->pos));
756 if(wrap) {
757 PosTwoDCoord su_geom = prjn->from->flat_geom;
758 TwoDCoord suc = su->pos;
759 suc.x += su_geom.x; // wrap around in x
760 prob += p_con * rnd_dist.Density(UnitDist(dist_type, prjn, ru->pos, suc));
761 suc.y += su_geom.y; // wrap around in x & y
762 prob += p_con * rnd_dist.Density(UnitDist(dist_type, prjn, ru->pos, suc));
763 suc.x = su->pos.x; // just y
764 prob += p_con * rnd_dist.Density(UnitDist(dist_type, prjn, ru->pos, suc));
765 suc = su->pos;
766 suc.x -= su_geom.x; // wrap around in x
767 prob -= p_con * rnd_dist.Density(UnitDist(dist_type, prjn, ru->pos, suc));
768 suc.y -= su_geom.y; // wrap around in y
769 prob += p_con * rnd_dist.Density(UnitDist(dist_type, prjn, ru->pos, suc));
770 suc.x = su->pos.x; // just y
771 prob += p_con * rnd_dist.Density(UnitDist(dist_type, prjn, ru->pos, suc));
772 }
773 return prob;
774}

778void PolarRndPrjnSpec::Connect_impl(Projection* prjn) {
779 if(!(bool)prjn->from) return;
780 if(same_seed)
781 rndm_seed.OldSeed();
782
783 int recv_no;
784 if(!self_con && (prjn->from.ptr() == prjn->layer))
785 recv_no = (int) ((p_con * (float)(prjn->from->units.leaves-1)) + .5f);
786 else
787 recv_no = (int) ((p_con * (float)prjn->from->units.leaves) + .5f);
788 if(recv_no <= 0) recv_no = 1;
789
790 // sending number is even distribution across senders plus some imbalance factor
791 float send_no_flt = (float)(prjn->layer->units.leaves * recv_no) / (float)prjn->from->units.leaves;
792 // add SEM as corrective factor
793 float send_sem = send_no_flt / sqrtf(send_no_flt);
794 int send_no = (int)(send_no_flt + 2.0f * send_sem + 5.0f); // polar needs some extra insurance
795 if(send_no > prjn->layer->units.leaves) send_no = prjn->layer->units.leaves;
796
797 // pre-allocate connections!
798 prjn->layer->RecvConsPreAlloc(recv_no, prjn);
799 prjn->from->SendConsPreAlloc(send_no, prjn);
800
801 Unit* ru, *su;
802 taLeafItr ru_itr;
803 PosTwoDCoord ru_geom = prjn->layer->flat_geom;
804 TwoDCoord ru_pos; // do this according to act_geom..
805 int cnt = 0;
806 for(ru = (Unit*)prjn->layer->units.FirstEl(ru_itr); ru;
807 ru = (Unit*)prjn->layer->units.NextEl(ru_itr), cnt++) {
808 ru_pos.y = cnt / ru_geom.x;
809 ru_pos.x = cnt % ru_geom.x;
810 RecvCons* recv_gp = NULL__null;
811 FloatTwoDCoord suc;
812 int n_con = 0;
813 int n_retry = 0;
814 while((n_con < recv_no) && (n_retry < max_retries)) { // limit number of retries
815 float dist = rnd_dist.Gen(); // just get random deviate from distribution
816 float angle = 2.0 * 3.14159265 * rnd_angle.Gen(); // same for angle
817 suc.x = dist * cos(angle);
818 suc.y = dist * sin(angle);
819 su = GetUnitFmOff(dist_type, wrap, prjn, ru_pos, suc);
820 if((su == NULL__null) || (!self_con && (ru == su))) {
821 n_retry++;
822 continue;
823 }
824 if(ru->ConnectFromCk(su, prjn, recv_gp))
825 n_con++;
826 else {
827 n_retry++; // already connected, retry
828 continue;
829 }
830 }
831 TestWarning(n_con < recv_no, "Connect_impl",
832 "target number of connections:",StringtaString(recv_no),
833 "not made, only made:",StringtaString(n_con));
834 }

2704void GaussRFPrjnSpec::Connect_impl(Projection* prjn) {
2705 if(!(bool)prjn->from) return;
2706 if(prjn->layer->units.leaves == 0) // an empty layer!
2707 return;
2708
2709 int n_cons = rf_width.Product();
2710 TwoDCoord rf_half_wd = rf_width / 2;
2711 TwoDCoord rug_geo = prjn->layer->flat_geom; // uses flat geom -- not ug based
2712 TwoDCoord su_geo = prjn->from->flat_geom;
2713
2714 TwoDCoord ruc;
2715 for(int alloc_loop=1; alloc_loop>=0; alloc_loop--) {
2716 for(ruc.y = 0; ruc.y < rug_geo.y; ruc.y++) {
2717 for(ruc.x = 0; ruc.x < rug_geo.x; ruc.x++) {
2718 Unit* ru_u = prjn->layer->UnitAtCoord(ruc);
2719 if(!ru_u) continue;
2720
2721 if(!alloc_loop)
2722 ru_u->RecvConsPreAlloc(n_cons, prjn);
2723
2724 TwoDCoord su_st;
2725 if(wrap) {
2726 su_st.x = (int)floor((float)ruc.x * rf_move.x) - rf_half_wd.x;
2727 su_st.y = (int)floor((float)ruc.y * rf_move.y) - rf_half_wd.y;
2728 }
2729 else {
2730 su_st.x = (int)floor((float)ruc.x * rf_move.x);
2731 su_st.y = (int)floor((float)ruc.y * rf_move.y);
2732 }
2733
2734 su_st.WrapClip(wrap, su_geo);
2735 TwoDCoord su_ed = su_st + rf_width;
2736 if(wrap) {
2737 su_ed.WrapClip(wrap, su_geo); // just wrap ends too
2738 }
2739 else {
2740 if(su_ed.x > su_geo.x) {
2741 su_ed.x = su_geo.x; su_st.x = su_ed.x - rf_width.x;
2742 }
2743 if(su_ed.y > su_geo.y) {
2744 su_ed.y = su_geo.y; su_st.y = su_ed.y - rf_width.y;
2745 }
2746 }
2747
2748
2749 TwoDCoord suc;
2750 TwoDCoord suc_wrp;
2751 for(suc.y = 0; suc.y < rf_width.y; suc.y++) {
2752 for(suc.x = 0; suc.x < rf_width.x; suc.x++) {
2753 suc_wrp = su_st + suc;
2754 if(suc_wrp.WrapClip(wrap, su_geo) && !wrap)
2755 continue;
2756 Unit* su_u = prjn->from->UnitAtCoord(suc_wrp);
2757 if(su_u == NULL__null) continue;
2758 if(!self_con && (su_u == ru_u)) continue;
2759
2760 ru_u->ConnectFrom(su_u, prjn, alloc_loop); // don't check: saves lots of time!
2761 }
2762 }
2763 }
2764 }
2765 if(alloc_loop) { // on first pass through alloc loop, do sending allocations
2766 prjn->from->SendConsPostAlloc(prjn);
2767 }
2768 }
