277void taiMisc::HandleScreenGeomChange_Window(const QRect& old_scrn_geom, QWidget* win) {
278 // if size bigger than avail size, then shrink it
279 // easiest policy is just to resize top-level wins by change in size
280 QRect r(win->frameGeometry());
281 QSize new_sz((int)(r.width() * ((float)scrn_s.width() / old_scrn_geom.width())),
282 (int)(r.height() * ((float)scrn_s.height() / old_scrn_geom.height())));
283 win->resize(new_sz);
284
285 //TODO
286 // move it to same size-relative point, in new coords
287 // this handles off-screen cases, as well as things like task bar redocks, etc.
288
289 // moves only if top left no longer accessible
290 const int marg = 50; // so user is able to grab title bar etc.
291 bool redo = false;
292 QPoint pos(r.topLeft());
293 if (r.x() > (scrn_geom.right() - marg)) {
294 pos.setX(scrn_geom.right() - marg);
295 redo = true;
296 } else if (r.x() < scrn_geom.left()) { // task bar left case
297 pos.setX(scrn_geom.left());
298 redo = true;
299 }
300 if (r.y() > (scrn_geom.bottom() - marg)) {
301 pos.setY(scrn_geom.bottom() - marg);
302 redo = true;
303 } else if (r.y() < scrn_geom.top()) { // task bar top case
304 pos.setY(scrn_geom.top());
305 redo = true;
306 }
307 if (redo) {
308 win->move(pos);
309 }

446void taiMisc::AdjustFont(int fontSpec, iFont& font) {
447 if (fontSpec & fonItalic) font.setItalic(true);
448 if (fontSpec & fonBold) font.setBold(true);
449 if (fontSpec & fonUnderline) font.setUnderline(true);
450 if ((fontSpec & fonStretch_mask) == fonSkinny) font.setStretch(QFont::Condensed);
451 if ((fontSpec & fonStretch_mask) == fonWide) font.setStretch(QFont::Expanded);
452 int fsz = (fontSpec & fonSize_mask);
453 switch (fsz) {
454 case fonSmall:
455 font.setPointSize(font.pointSize - FONT_SM);
456 break;
457 //NOTE: assumes not using pixel-size, otherwise
458 case fonMedium:
459 font.setPointSize(font.pointSize - FONT_MED);
460 case fonBig: // BIG is the default, no adjust needed
461 default: //defFontSize
462 break;
463 }
464}

467 iSize rval;

475 break;
476 case dlgSmall:

478 else rval = iSize((scrn_s.w * 2) / 5, (scrn_s.h * 3) / 10); // .4w .3h small horizontally oriented dialog
479 break;
480 case dlgBig:
481 if (ver) rval = iSize((scrn_s.w * 2) / 5, (scrn_s.h * 4) / 5);// .4w .8h big vertically oriented dialog
482 else rval = iSize((scrn_s.w * 4) / 5, (scrn_s.h * 3) / 5);// .8w .6h big horizontally oriented dialog
483 break;
484 case dlgMedium:
485 default: // medium, the default
486 if (ver) rval = iSize((scrn_s.w * 2) / 5, (scrn_s.h * 3) / 5);// .4w .6h medium vertically oriented dialog
487 else rval = iSize((scrn_s.w * 3) / 5, (scrn_s.h * 2) / 5);// .6w .4h medium horizontally oriented dialog

977void taiMisc::KeepInView_SA(QAbstractScrollArea* sa, QWidget* sa_main_widg, QWidget* widg) {
978 int w_ht = widg->height();
979 int top_in_vc = MapToAreaV_SA(sa, sa_main_widg, widg, 0);
980 int ctr_pos = top_in_vc + w_ht / 2;
981 int bot_pos = top_in_vc + w_ht;
982
983 if(PosInView_SA(sa, top_in_vc) && PosInView_SA(sa, bot_pos))
984 return; // already in view
985
986 int vpt_ht = sa->viewport()->height();
987 int scpos = sa->verticalScrollBar()->value();
988 int scbot = scpos + sa->viewport()->height();
989 if(ctr_pos < scpos) { // closer to top
990 int nwtop = MAX(top_in_vc-12,0)(((top_in_vc-12) > (0)) ? (top_in_vc-12) : (0));
991 ScrollTo_SA(sa, nwtop); // scroll up to top, plus a bit of margin
992 taMisc::Info("Keep top:", StringtaString(nwtop));
993 }
994 else {
995 int nwbot = MIN(bot_pos+12,vpt_ht)(((bot_pos+12) < (vpt_ht)) ? (bot_pos+12) : (vpt_ht));
996 int nwtop = MAX(nwbot - vpt_ht, 0)(((nwbot - vpt_ht) > (0)) ? (nwbot - vpt_ht) : (0));
997 ScrollTo_SA(sa, nwtop); // scroll down to bottom plus some margin
998 taMisc::Info("Keep bot:", StringtaString(nwbot));
999 }
1000}
