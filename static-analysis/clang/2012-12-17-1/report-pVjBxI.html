136StringtaString TemtClient::NextToken(const StringtaString& str, int& p, bool& err) {
137 int max = str.length() - 1;
138 // skip ws
139 char c;
140 while ((p <= max) && !err) {
141 c = str[p];
142 if (!isspace(c)) break;
143 ++p;
144 }
145 if (p > max) return _nilStringtaString(); // end
146
147 // look for " and put us in quoted mode if so
148 c = str[p];
149 if (c == '\"')
150 return ReadQuotedString(str, p, err);
151
152 // single = is a token
153 if (c == '=') {
154 ++p;
155 return "=";
156 }
157
158 // ok, read a raw token, which is chars up to ws or =
159 StringtaString rval;
160 while (p <= max) {
161 c = str[p];
162 if ((c == '=') || isspace(c)) break;
163 rval.cat(c);
164 ++p;
165 }
166 return rval;
167}

289 prog->Run();

297 SendError("RunProgram " + pnm + "->Run() failed with ret_val: "

300 }
301 } else { // async
302 adapter()->SetProg(prog);
303 QTimer::singleShot(0, adapter(), SLOT(prog_Run())"1""prog_Run()");

369bool TemtClient::TableParams::ValidateParams(TemtClient::TableParams::Cmd cmd, bool mat_) {
370 // defaults for values -- some of the guys below alter these, before we read them
371 rows = 1;
372 header = false;
373 markers = false;
374
375 // cmd decodes
376 bool get = false;
377 bool get_set = false;
378 bool remove = false;
379 bool append = false;
380 bool is_cell = false;
381 switch (cmd) {
382 case TemtClient::TableParams::Get:
383 get = true;
384 get_set = true;
385 break;
386 case TemtClient::TableParams::Append:
387 append = true;
388 header = true;
389 break;
390 case TemtClient::TableParams::Remove:
391 rows = -1;
392 remove = true;
393 break;
394 case TemtClient::TableParams::Set:
395 get_set = true;
396 break;
397 case TemtClient::TableParams::Cell:
398 is_cell = true;
399 mat = mat_;
400 cell = 0;
401 break;
402 }
403
404 // cell guys are pos params, so do them, and bail
405 // also, they use uniony guys
406 if (is_cell) {
407 bool ok = true;
408 // get col, row, and cell params
409 col = tc->pos_params.SafeEl(1).toInt(&ok);
410 if (col < 0) col = tab->cols() + col;
411 // check in bounds
412 if (!ok || (col < 0) || (col >= tab->cols())) {
413 tc->SendError("col '" + StringtaString(col) + "' out of bounds");
414 return false;
415 }
416
417 row = tc->pos_params.SafeEl(2).toInt(&ok);
418 if (row < 0) row = tab->rows + row;
419 if (!ok || (row < 0) || (row >= tab->rows)) {
420 tc->SendError("row '" + StringtaString(row) + "' out of bounds");
421 return false;
422 }
423
424 if (mat)
425 cell = tc->pos_params.SafeEl(3).toInt(&ok);
426 if (!ok || (cell < 0) || (cell >= tab->data.FastEl(col)->cell_geom.Product())) {
427 tc->SendError("cell '" + StringtaString(cell) + "' out of bounds");
428 return false;
429 }
430 return true;
431 }
432
433 // all the possible params -- not all defined for all cmds
434 rows = tc->name_params.GetValDef("rows", rows).toInt();
435 row_from = tc->name_params.GetValDef("row_from", 0).toInt();
436 row_to = tc->name_params.GetValDef("row_to", -1).toInt();
437 col_from = tc->name_params.GetValDef("col_from", 0).toInt();
438 col_to = tc->name_params.GetValDef("col_to", -1).toInt();
439 header = tc->name_params.GetValDef("header", header).toBool();
440 markers = tc->name_params.GetValDef("markers", markers).toBool();
441
442 // init vals that need to be correctly set even if validation fails
443 if (header) lines = 1; else lines = 0;
444
445 // validate and normalize the params
446 if (row_from < 0) row_from = tab->rows + row_from;
447 if (row_to < 0) row_to = tab->rows + row_to;
448 if (get_set) {
449 if ((row_from < 0) || (row_from >= tab->rows)) {
450 tc->SendError("row_from '" + StringtaString(row_from) + "' out of bounds");
451 return false;
452 }
453
454 if ((row_to < 0) || (row_to >= tab->rows)) {
455 tc->SendError("row_to '" + StringtaString(row_to) + "' out of bounds");
456 return false;
457 }
458 }
459
460 if (append) {
461 if (rows < 0) {
462 tc->SendError("rows '" + StringtaString(rows) + "' out of bounds");
463 return false;
464 }
465 row_from = tab->rows;
466 row_to = (row_from + rows) - 1;
467 }
468
469 if (remove) {
470 // note: only ok for row_from to be out of bounds if empty
471 if ((row_from < 0) && (tab->rows > 0)) {
472 tc->SendError("row_from '" + StringtaString(row_from) + "' out of bounds");
473 return false;
474 }
475 if (rows < 0) {
476 rows = tab->rows;
477 }
478 // adjust rows to be true rows
479 if ((row_from + rows) > tab->rows)
480 rows = tab->rows - row_from;
481 return true; // skip all the other stuff
482 }
483
484 // make sure lines is valid now, before possibly bailing on other conditions
485 lines += (row_to - row_from) + 1;
486
487 if (col_from < 0) col_from = tab->cols() + col_from;
488 if ((col_from < 0) || (col_from >= tab->cols())) {
489 tc->SendError("col_from '" + StringtaString(col_from) + "' out of bounds");
490 return false;
491 }
492
493 if (col_to < 0) col_to = tab->cols() + col_to;
494 if ((col_to < 0) || (col_to >= tab->cols())) {
495 tc->SendError("col_to '" + StringtaString(col_to) + "' out of bounds");
496 return false;
497 }
498
499 return true;
500}

502void TemtClient::cmdAppendData() {
503 StringtaString tnm = pos_params.SafeEl(0);
504 DataTable* tab = GetAssertTable(tnm);
505 if (!tab) return;
506 // note: ok if running
507
508 TableParams p(this, tab);
509 bool cmd_ok = p.ValidateParams(TemtClient::TableParams::Append);
510 //NOTE: p will have already sent an ERROR if cmd_ok is false
511 // any subsequenct failure must send an ERROR
512
513
514 // ok, now we must block/waiting until all expected lines received
515 // note: some of the expected lines may already be in lines buffer
516 setState(CS_DATA_IN);
517
518 while ((lines.size < p.lines) && (state != CS_DISCONNECTED)) {
519 // note: signals not invoked again inside this event loop
520 if (sock->canReadLine())
521 sock_readyRead();
522 else
523 sock->waitForReadyRead();
524 }
525
526 if (state == CS_DISCONNECTED)
527 return;
528
529 setState(CS_READY);
530
531 if (!cmd_ok) { // error of some occurred, so we just accepted lines
532 // must remove first n lines
533 while ((p.lines > 0) && (lines.size > 0)) {
534 lines.RemoveIdx(0);
535 p.lines--;
536 }
537 return;
538 }
539
540 // trivial case with no lines
541 if (p.lines == 0) goto ok_exit;
542
543 {
544
545 //note: we must have enough lines to have made it here
546 tab->StructUpdate(true);
547 int ln_num = 0; // to do header/init stuff
548 while ((p.lines > 0) && (lines.size > 0)) {
549 StringtaString ln;
550 //note: parser needs the eol, so we have to add it back in
551 ln = lines.FastEl(0) + '\n';
552 lines.RemoveIdx(0);
553 p.lines--;
554 // simplest way to be compat with arcane DataTable load api
555 // is to tack the markers on if we aren't expecting them
556 if (!p.markers) {
557 if (p.header && (ln_num == 0))
558 ln = "_H:\t" + ln;
559 else ln = "_D:\t" + ln;
560 }
561 // note: making a new one each loop simplifies things
562 istringstream istr(ln.chars());
563
564 tab->LoadDataRowEx_strm(istr, DataTable::TAB,
565 true, // quote_str
566 (ln_num == 0) // clear load schema on first guy
567 );
568 ++ln_num;
569 }
570 tab->StructUpdate(false);
571 }
572ok_exit:
573 // ok, we did!
574 SendOk();
575}

578 StringtaString tnm = pos_params.SafeEl(0);
579 DataTable* tab = GetAssertTable(tnm);
580 if (!tab) return;
581 // note: ok if running
582

598

604 true, // quote_str
605 p.markers, // row_mark
606 p.col_from, p.col_to);

608 Write(ln);

932void TemtClient::sock_readyRead() {
933 if (!sock) return;
934 // we only do lines -- will call us again when a line is ready
935 if (!sock->canReadLine()) return;
936
937 QByteArray ba;
938 StringtaString line;
939 // need to keep fetching lines, since we only get notified once
940 // note: typical case is we get one full line at a time
941 bool ok = true; // in case reading goes daft
942 while (ok && sock->canReadLine()) {
943 ba = sock->readLine();
944 qint64 line_len = ba.size();
945 if (line_len <= 0) {
946 ok = false;
947 break;
948 }
949 line.set(ba.data(), (int)line_len); // note: includes \n or platform separator
950 // strip cr/lf chars for client platform neutrality
951 line.gsub('\n', "");
952 line.gsub('\r', "");
953 lines.Add(line);
954 }
955 HandleLines();
956}

958void TemtClient::HandleLines() {
959 // so, we've grabbed as many lines as we can now, so what do we do???
960 // cases:
961 // DATA_IN: if all received, then dispatch handler, else return
962 // READY: we can pull first guy, and do a command
963 while (lines.size > 0) {
964 switch (state) {
965 case CS_READY: {
966 StringtaString line = lines.FastEl(0);
967 lines.RemoveIdx(0);
968 ParseCommand(line);
969 } break;
970 case CS_DATA_IN: {
971 // we would have reached here re-entrantly, within a cmd, so just exit
972 return;
973 } break;
974 case CS_DISCONNECTED: {
975 //shouldn't happen!!!
976 } break;
977 //default:
978 // handle all cases explicitly!
979 };
980 }

1097 delete server;

1106 taMisc::server_active = true;

1111 // setup data writing for hello or error block
1112 QByteArray block;
1113 QDataStream out(&block, (QIODevice::OpenMode)QIODevice::WriteOnly);
1114 out.setVersion(QDataStream::Qt_4_0);
1115
1116 // get the latest connection, and always set it to self-destruct on close
1117 QTcpSocket* ts = server->nextPendingConnection();
1118 QObject::connect(ts, SIGNAL(disconnected())"2""disconnected()",
1119 ts, SLOT(deleteLater())"1""deleteLater()");
1120 // we only allow 1 client (in this version of pdp), so refuse others
1121 if (clients.size >= 1) {
1122 out << "ERROR 1 : too many connections already, closing...\n";
1123 ts->write(block);
1124 ts->disconnectFromHost();
1125 return;
1126 }
1127 TemtClient* cl = (TemtClient*)clients.New(1);
1128 cl->server = this;
1129 cl->SetSocket(ts);
1130
1131 StringtaString banner = "Emergent Server v" + taMisc::version + "\n";
1132 out << banner.chars();
1133 ts->write(block);
1134 DataChanged(DCR_ITEM_UPDATED);
