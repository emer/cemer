463void cssPtr::operator=(const cssEl& s) {
464 if((s.GetType() == T_Ptr) || (s.GetType() == T_Array)) {
465 cssPtr* tmp = (cssPtr*)s.GetNonRefObj();
466 cssEl* s_type = tmp->GetNonPtrTypeObj();
467 cssEl* nptr_type = GetNonPtrTypeObj();
468
469 // if it's a void pointer, use type of what it points to
470 // if it's a class inst, get its class type
471 if(s_type->GetType() == T_Void)
472 s_type = tmp->ptr.El();
473 if(s_type->GetType() == T_Class)
474 s_type = ((cssClassInst*)s_type->GetNonRefObj())->type_def;
475
476 // if either one's a Void, then assignment is automatically ok
477 bool auto_ok = false;
478 if((nptr_type->GetType() == T_Void) || (s_type->GetType() == T_Void))
479 auto_ok = true;
480 if(!auto_ok) {
481 if(nptr_type->GetType() != s_type->GetType()) {
482 cssMisc::Error(prog, "Pointer type mismatch between:",nptr_type->GetTypeName(),"and",
483 s_type->GetTypeName());
484 return;
485 }
486 // if we have class types, check for inheritance
487 // allow both up and down casting
488 else if(nptr_type->GetType() == T_ClassType) {
489 cssClassType* ths_class = (cssClassType*) nptr_type->GetNonRefObj();
490 cssClassType* s_class = (cssClassType*) s_type->GetNonRefObj();
491 if(!ths_class->InheritsFrom(s_class) &&
492 !s_class->InheritsFrom(ths_class)) {
493 cssMisc::Error(prog, "Pointer type mismatch between:",nptr_type->GetTypeName(),"and",
494 s_type->GetTypeName());
495 return;
496 }
497 }
498 }
499
500 // assignment is now type-correct if you get here
501 // if its an array, we refer to it directly, not to its pointer
502 if(s.GetType() == T_Array)
503 SetPtr(tmp->GetAddr());
504 else
505 SetPtr(tmp->ptr);
506 }
507 else {
508 int sval = (Int)s;
509 if(sval == 0)
510 SetPtr(cssMisc::VoidElPtr);
511 else
512 cssMisc::Error(prog, "Assigning pointer to non-ptr value");
513 }
514}

787 if(c != '[') {
788 taMisc::Error("Did not get array size");
789 taMisc::skip_past_err(strm);
790 return;
791 }
792 strm.get();

794 if(c != ']') {
795 taMisc::Error("Did not get array size");
796 taMisc::skip_past_err(strm);
797 return;
798 }
799 int sz = (int)taMisc::LexBuf;
800 Alloc(sz);

803 if(c != '{') {
804 taMisc::Error("Did not get array starting bracket");
805 taMisc::skip_past_err(strm);
806 return;
807 }
808 strm.get(); // get the bracket
809 int i;
810 for(i=0; i<items->size; i++) {
811 cssEl* itm = items->FastEl(i);
812 itm->Load(strm);
813 if(strm.peek() == ';') strm.get(); // skip past ending semi
814 if(strm.peek() == '}')
815 break;
816 }
817 c = taMisc::read_till_rbracket(strm, true);
818 if(c != '}') {
819 taMisc::Error("Did not get array terminator '}'");
820 taMisc::skip_past_err(strm);
821 return;
822 }
823 strm.get(); // get rb
824 if(strm.peek() == ';') strm.get(); // skip past ending semi
825}

1288void cssClassType::CallVoidMethod(cssClassInst* tok, const StringtaString& meth_nm) {
1289 StringtaString nm = meth_nm;
1290 if(nm.contains(')')) {
1291 nm = nm.before(')');
1292 nm = nm.after('(');
1293 }
1294 cssElPtr ptr = methods->FindName(nm); // find name of this type..
1295 if(ptr == 0) return; // didn't find it..
1296 cssMbrScriptFun* meth = (cssMbrScriptFun*)methods->FastEl(ptr.dx);
1297 if(meth->GetSubProg() != NULL__null) {
1298 meth->GetSubProg()->state |= cssProg::State_NoBreak; // do not break this guy
1299 }
1300
1301 // get the appropriate context to run in: either token top
1302 // or the generic top-level
1303 cssProg* prg = cssMisc::CDtorProg;
1304 prg->state |= cssProg::State_IsTmpProg; // flag to cssProgSpace::Cont to not run past this guy
1305 prg->state |= cssProg::State_NoBreak;
1306 cssProgSpace* old_tok_top = NULL__null;
1307 cssProgSpace* old_prg_top = NULL__null;
1308 if((tok->prog != NULL__null) && (tok->prog->top != NULL__null)) {
1309 // todo: this should not be needed anymore
1310 cssMisc::SetCurTop(tok->prog->top); // reparent to current top
1311 old_prg_top = prg->SetTop(cssMisc::cur_top);
1312 }
1313 else {
1314 cssMisc::SetCurTop(cssMisc::Top);
1315 old_prg_top = prg->SetTop(cssMisc::cur_top);
1316 tok->prog = prg;
1317 }
1318
1319 cssEl::Ref(tok); // make sure it doesn't get to 0 refcount in arg del..
1320 cssEl::Ref(meth);
1321
1322 prg->Reset();
1323 // not using prog::code function because it might go to another top..
1324 cssElPtr elp; elp.SetDirect(meth); // code the meth
1325 cssInst* tmp = new cssInst(prg, elp);
1326 tmp->idx = prg->AddCode(tmp);
1327
1328 prg->Restart();
1329 prg->top->Shove(prg);
1330 prg->Stack()->Push(&cssMisc::Void); // argstop
1331 prg->Stack()->Push(tok);
1332 prg->top->Cont(); // runs it
1333 prg->top->Pull(); // pull the meth/dtor off of stack!
1334 prg->Reset();
1335
1336 if(meth->GetSubProg() != NULL__null) {
1337 meth->GetSubProg()->state &= ~cssProg::State_NoBreak; // do not break this guy
1338 }
1339
1340 cssEl::unRef(tok); // undo that
1341 cssEl::unRef(meth); // undo that
1342
1343 if(old_prg_top != NULL__null) prg->PopTop(old_prg_top);
1344 if(old_tok_top != NULL__null) tok->prog->PopTop(old_prg_top);
1345 cssMisc::PopCurTop();
1346}

1892void cssClassInst::Load(istream& strm) {
1893 int c;
1894 c = taMisc::read_till_lbracket(strm,false); // get the lbrack
1895 if(c == EOF(-1)) return;
1896 if(c != '{') {
1897 taMisc::Error("*** left bracket not found for start of class");
1898 taMisc::skip_past_err(strm);
1899 return;
1900 }
1901 StringtaString tp_nm = taMisc::LexBuf;
1902 tp_nm = tp_nm.before(' ');
1903 cssElPtr tpptr = prog->top->FindTypeName(tp_nm);
1904 if((tpptr == 0) || (tpptr.El()->GetType() != T_ClassType) ||
1905 !type_def->InheritsFrom((cssClassType*)tpptr.El())) {
1906 taMisc::Error("*** type name:",tp_nm,"not a valid type, or this type:",
1907 type_def->name, "does not inherit from it");
1908 taMisc::skip_past_err(strm);
1909 return;
1910 }
1911
1912 do {
1913 c = taMisc::read_word(strm, true);
1914 if(c == EOF(-1)) return;
1915 if(c == '}') {
1916 strm.get();
1917 if(strm.peek() == ';') strm.get(); // skip past ending semi
1918 return;
1919 }
1920
1921 StringtaString mb_name = taMisc::LexBuf;
1922 if((c == ' ') || (c == '\n') || (c == '\t')) { // skip past white
1923 strm.get();
1924 c = taMisc::skip_white(strm, true);
1925 }
1926 if(c != '=') {
1927 taMisc::Error("*** did not get '=' for member");
1928 taMisc::skip_past_err(strm);
1929 continue;
1930 }
1931 else {
1932 strm.get(); // get the equals sign
1933 c = taMisc::skip_white(strm, true); // then skip past any new whitespace
Value stored to 'c' is never read
1934 }
1935
1936 cssElPtr mb_ptr = members->FindName(mb_name);
1937 if(mb_ptr == 0) {
1938 taMisc::Error("*** Member:", mb_name, "not found in type:", type_def->name);
1939 taMisc::skip_past_err(strm);
1940 continue;
1941 }
1942 cssEl* mb = mb_ptr.El();
1943
1944 mb->Load(strm);
1945
1946 } while(1);
1947}
