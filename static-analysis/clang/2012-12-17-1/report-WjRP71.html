440 int flags = iTreeViewItem::DNF_IS_FOLDER | iTreeViewItem::DNF_NO_CAN_DROP;
441 switch (tik) {

443 }
444 break;
445 case taMisc::TIK_MEMBER: {

447 break;
448 case taMisc::TIK_PROPERTY: {

460 break;
461 }

463 // if we are an enum, we handle differently (only showing enums, and directly under us)

466 StringtaString tree_nm;
467 for (int j = 0; j < td->enum_vals.size; ++j) {

471

473
474 if (dl == NULL__null) continue; // shouldn't happen...
475
476 tree_nm = dl->GetDisplayName();
477 if (tree_nm.empty()) {
478 //TODO tree_nm = link()->AnonymousItemName(typ->name, i);

481 last_child_node = dl->CreateTreeDataNode((MemberDef*)NULL__null, this,
482 last_child_node, tree_nm, flags_j);
483 }
484 break;
485 }

489 if (td->HasEnumDefs()) {

491 taTypeSpaceDataLink* tsdl = static_cast<taTypeSpaceDataLink*>(
492 taiViewType::StatGetDataLink(ts, &TA_TypeSpace));

497

499 if (td->HasSubTypes()) {

501 taTypeSpaceDataLink* tsdl = static_cast<taTypeSpaceDataLink*>(
502 taiViewType::StatGetDataLink(st, &TA_TypeSpace));
503 last_child_node = tsdl->CreateTreeDataNode(
504 NULL__null, this, last_child_node, "sub types", flags);
505 tsdl->dm = taTypeSpaceDataLink::DM_DefaultSubTypes;
506 }
507 // members -- note: don't sort, since they are in a programmer order already
508 if (td->members.size > 0) {
509 dl = taiViewType::StatGetDataLink(&td->members, &TA_MemberSpace);
510 last_child_node = dl->CreateTreeDataNode(

513

516 dl = taiViewType::StatGetDataLink(&td->methods, &TA_MethodSpace);

661 if (!tsdl->ShowChild(td)) return false;
662 if (!ShowItem(td)) return false;
663 // basic behavior is that we don't show derivitive types, ex. consts, refs, ptrs, etc.
664 if ((td->ptr > 0)
665 || (td->ref)
666 || (td->formal)
667 ) return false;
668 // need to check parentage for const -- (note: const is not formal for some weird reason)
669 if (td->InheritsFrom(&TA_const))
670 return false;
671
672 return true;
673}

881taiDataLink* taTypeInfoViewType::GetDataLink(void* el, TypeDef* td) {
882 if (!el) return NULL__null;
883
884 taMisc::TypeInfoKind tik = taMisc::TypeToTypeInfoKind(td);
885
886 switch (tik) {
887 case taMisc::TIK_ENUM:
888 case taMisc::TIK_MEMBER:
889 case taMisc::TIK_PROPERTY:
890 case taMisc::TIK_METHOD:
891 case taMisc::TIK_TYPE: {
892 TypeItem* ti = static_cast<TypeItem*>(el);
893 if (ti->data_link) return static_cast<taiDataLink*>(ti->data_link);
894 else return new taTypeInfoDataLink(tik, ti);
895 }
896 case taMisc::TIK_MEMBERSPACE: {
897 MemberSpace* s = static_cast<MemberSpace*>(el);
898 if (s->data_link != NULL__null) return static_cast<taiDataLink*>(s->data_link);
899 else return new taMemberSpaceDataLink(s);
900 }
901 case taMisc::TIK_PROPERTYSPACE: {
902 PropertySpace* s = static_cast<PropertySpace*>(el);
903 if (s->data_link != NULL__null) return static_cast<taiDataLink*>(s->data_link);
904 else return new taPropertySpaceDataLink(s);
905 }
906 case taMisc::TIK_METHODSPACE: {
907 MethodSpace* s = static_cast<MethodSpace*>(el);
908 if (s->data_link != NULL__null) return static_cast<taiDataLink*>(s->data_link);
909 else return new taMethodSpaceDataLink(s);
910 }
911 case taMisc::TIK_TYPESPACE: {
912 TypeSpace* s = static_cast<TypeSpace*>(el);
913 if (s->data_link != NULL__null) return static_cast<taiDataLink*>(s->data_link);
914 else return new taTypeSpaceDataLink(s);
915 }
916 case taMisc::TIK_TOKENSPACE:
917 default:
918 return NULL__null;
919 }
920 return NULL__null; //compiler food
