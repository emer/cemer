246 case DIV:

419static double gser(double a, double x) {
420 int n;
421 double gln,sum,del,ap;
422
423 gln=taMath_double::gamma_ln(a);
424 if (x <= 0.0) {
425 if (x < 0.0) { fprintf(stderrstderr, "x < 0 in gser()\n"); return 0; }
426 return 0;
427 }
428 else {
429 ap=a;
430 del=sum=1.0/a;
431 for (n=1;n<=taMath_double::max_iterations;n++) {
432 ap += 1.0;
433 del *= x/ap;
434 sum += del;
435 if (fabs(del) < fabs(sum)*taMath_double::err_tolerance)
436 return sum*exp(-x+a*log(x)-(gln));
437 }
438 fprintf(stderrstderr, "a too large, max_iterations too small in gser()\n");
439 return 0;
440 }
441}

443static double gcf(double a, double x) {
444 int n;
445 double gln;
446 double gold=0.0,g,fac=1.0,b1=1.0;
447 double b0=0.0,anf,ana,an,a1,a0=1.0;
448
449 gln=taMath_double::gamma_ln(a);
450 a1=x;
451 for (n=1;n<=taMath_double::max_iterations;n++) {
452 an=(double) n;
453 ana=an-a;
454 a0=(a1+a0*ana)*fac;
455 b0=(b1+b0*ana)*fac;
456 anf=an*fac;
457 a1=x*a0+anf*a1;
458 b1=x*b0+anf*b1;
459 if (a1) {
460 fac=1.0/a1;
461 g=b1*fac;
462 if (fabs((g-gold)/g) < taMath_double::err_tolerance)
463 return exp(-x+a*log(x)-(gln))*g;
464 gold=g;
465 }
466 }
467 fprintf(stderrstderr, "a too large, max_iterations too small in gcf()\n");
468 return 0;
469}

471double taMath_double::gamma_ln(double z) {
472 double x,tmp,ser; /* make sure double-precision.. */
473 static double cof[6]={ 76.18009173, -86.50532033, 24.01409822,
474 -1.231739516, 0.120858003e-2, -0.536382e-5 };
475 int j;
476
477 x=z-1.0;
478 tmp=x+5.5;
479 tmp -= (x+0.5)*log(tmp);
480 ser=1.0;
481 for (j=0;j<=5;j++) {
482 x += 1.0;
483 ser += cof[j]/x;
484 }
485 return -tmp+log(2.50662827465*ser);
486}

505/**********************************
506 beta function
507***********************************/
508
509static double betacf_double(double a, double b, double x) {
510 double qap,qam,qab,em,tem,d;
511 double bz,bm=1.0,bp,bpp;
512 double az=1.0,am=1.0,ap,app,aold;
513 int m;
514
515 qab=a+b;
516 qap=a+1.0;
517 qam=a-1.0;
518 bz=1.0-qab*x/qap;
519 for (m=1;m<=taMath_double::max_iterations;m++) {
520 em=(double) m;
521 tem=em+em;
522 d=em*(b-em)*x/((qam+tem)*(a+tem));
523 ap=az+d*am;
524 bp=bz+d*bm;
525 d = -(a+em)*(qab+em)*x/((qap+tem)*(a+tem));
526 app=ap+d*az;
527 bpp=bp+d*bz;
528 aold=az;
529 am=ap/bpp;
530 bm=bp/bpp;
531 az=app/bpp;
532 bz=1.0;
533 if (fabs(az-aold) < (taMath_double::err_tolerance*fabs(az))) return az;
534 }
535 fprintf(stderrstderr, "a or b too big, or max_iterations too small in betacf()\n");
536 return 0;
537}

543double taMath_double::beta_i(double a, double b, double x) {
544 double bt;
545
546 if (x < 0.0 || x > 1.0) { fprintf(stderrstderr, "Bad x in beta_i()\n"); return 0; }
547 if (x == 0.0 || x == 1.0) bt=0.0;
548 else
549 bt=exp(gamma_ln(a+b)-gamma_ln(a)-gamma_ln(b)+a*log(x)+b*log(1.0-x));
550 if (x < (a+1.0)/(a+b+2.0))
551 return bt*betacf_double(a,b,x)/a;
552 else
553 return 1.0-bt*betacf_double(b,a,1.0-x)/b;
554}

699 do {
700 do {
701 do {
702 v1=2.0*MTRnd::genrand_res53()-1.0;
703 v2=2.0*MTRnd::genrand_res53()-1.0;
704 } while (v1*v1+v2*v2 > 1.0);
705 y=v2/v1;
706 am=ia-1;
707 s=sqrt(2.0*am+1.0);
708 x=s*y+am;
709 } while (x <= 0.0);
710 e=(1.0+y*y)*exp(am*log(x/am)-s*y);
711 } while (MTRnd::genrand_res53() > e);

2352 int idx = i + j - off;
2353 if(idx < 0 || idx >= in_vec->size) {
2354 dnorm += kernel->FastEl(j);
2355 }
2356 else {
2357 sum += in_vec->FastEl(idx) * kernel->FastEl(j);
2358 }
2359 }
2360 if(dnorm > 0.0 && dnorm < 1.0) { // renorm
2361 sum /= (1.0 - dnorm);
2362 }
2363 out_vec->FastEl(i) = sum;
2364 }
2365 }
2366 else {
2367 if(in_vec->size < kernel->size) {
2368 taMisc::Error("vec_convolve: input vector size < kernel size -- cannot convolve");
2369 return false;
2370 }
2371 if(out_vec->size != in_vec->size - kernel->size)
2372 out_vec->SetGeom(1, in_vec->size - kernel->size);
2373 for(int i=0;i<out_vec->size;i++) {
2374 double sum = 0.0;
2375 for(int j=0;j<kernel->size;j++) {
2376 sum += in_vec->FastEl(i+j) * kernel->FastEl(j);
2377 }
2378 out_vec->FastEl(i) = sum;
2379 }
2380 }
2381 return true;
2382}

2384bool taMath_double::vec_kern2d_gauss(double_Matrix* kernel, int sz_x, int sz_y,
2385 double sigma_x, double sigma_y) {

2398 }

2597bool taMath_double::mat_mds_owrite(double_Matrix* a, double_Matrix* xy_coords, int x_axis_c, int y_axis_c) {

2599 taMisc::Error("*** mat_mds: matrix is not 2 dimensional!");
2600 return false;
2601 }
2602 if(a->dim(0) != a->dim(1)) {
2603 taMisc::Error("*** mat_mds: matrix is not square!");
2604 return false;
2605 }
2606 int dim = a->dim(0);
2607
2608 if((x_axis_c < 0) || (x_axis_c >= dim)) {
2609 taMisc::Error("*** mat_mds: x_axis component must be between 0 and",StringtaString(dim-1));
2610 return false;
2611 }
2612 if((y_axis_c < 0) || (y_axis_c >= dim)) {
2613 taMisc::Error("*** mat_mds: y_axis component must be between 0 and",StringtaString(dim-1));
2614 return false;
2615 }
2616
2617 // first square the individual elements
2618 mat_mult_els(a, a);
2619
2620 // then double-center the matrix
2621 for(int i=0; i<dim; i++) {
2622 double sum = 0.0;
2623 for(int j=0; j<dim; j++)
2624 sum += a->FastEl(i, j);
2625 sum /= (double)dim;
2626 for(int j=0; j<dim; j++)
2627 a->FastEl(i, j) -= sum;
2628 }
2629 for(int j=0; j<dim; j++) {
2630 double sum = 0.0;
2631 for(int i=0; i<dim; i++)
2632 sum += a->FastEl(i, j);
2633 sum /= (double)dim;
2634 for(int i=0; i<dim; i++)
2635 a->FastEl(i, j) -= sum;
2636 }
2637
2638 for(int i=0;i<a->size;i++) a->FastEl_Flat(i) *= -.5;
2639
2640 double_Matrix evals(false);
2641 double_Matrix evecs(false);
2642 mat_eigen_owrite(a, &evals, &evecs);
2643
2644 xy_coords->SetGeom(2, 2, dim);
2645 // multiply the eigenvectors by sqrt(eigen values)
2646 double evsq = taMath_double::sqrt(taMath_double::fabs(evals.FastEl(x_axis_c)));
2647 for(int i=0;i<dim;i++) {
2648 float val = evecs.FastEl(i, x_axis_c) * evsq;
2649 xy_coords->FastEl(0, i) = val; // 0 = x coord
2650 }
2651
2652 evsq = taMath_double::sqrt(taMath_double::fabs(evals.FastEl(y_axis_c)));
2653 for(int i=0;i<dim;i++) {
2654 float val = evecs.FastEl(i, y_axis_c) * evsq;
2655 xy_coords->FastEl(1, i) = val; // 1 = y coord
2656 }
2657
2658 return true;
2659}

3691static float gser(float a, float x) {
3692 int n;
3693 float gln,sum,del,ap;
3694
3695 gln=taMath_float::gamma_ln(a);
3696 if (x <= 0.0) {
3697 if (x < 0.0) { fprintf(stderrstderr, "x < 0 in gser()\n"); return 0; }
3698 return 0;
3699 }
3700 else {
3701 ap=a;
3702 del=sum=1.0/a;
3703 for (n=1;n<=taMath_float::max_iterations;n++) {
3704 ap += 1.0;
3705 del *= x/ap;
3706 sum += del;
3707 if (fabs(del) < fabs(sum)*taMath_float::err_tolerance)
3708 return sum*exp(-x+a*log(x)-(gln));
3709 }
3710 fprintf(stderrstderr, "a too large, max_iterations too small in gser()\n");
3711 return 0;
3712 }

3715static float gcf(float a, float x) {
3716 int n;
3717 float gln;
3718 float gold=0.0,g,fac=1.0,b1=1.0;
3719 float b0=0.0,anf,ana,an,a1,a0=1.0;
3720
3721 gln=taMath_float::gamma_ln(a);
3722 a1=x;
3723 for (n=1;n<=taMath_float::max_iterations;n++) {
3724 an=(float) n;
3725 ana=an-a;
3726 a0=(a1+a0*ana)*fac;
3727 b0=(b1+b0*ana)*fac;
3728 anf=an*fac;
3729 a1=x*a0+anf*a1;
3730 b1=x*b0+anf*b1;
3731 if (a1) {
3732 fac=1.0/a1;
3733 g=b1*fac;
3734 if (fabs((g-gold)/g) < taMath_float::err_tolerance)
3735 return exp(-x+a*log(x)-(gln))*g;
3736 gold=g;
3737 }
3738 }
3739 fprintf(stderrstderr, "a too large, max_iterations too small in gcf()\n");
3740 return 0;
3741}

3746 -1.231739516f, 0.120858003e-2f, -0.536382e-5f };

3778 beta function
3779***********************************/
3780
3781static float betacf_float(float a, float b, float x) {

3783 float bz,bm=1.0,bp,bpp;
3784 float az=1.0,am=1.0,ap,app,aold;
3785 int m;
3786
3787 qab=a+b;
3788 qap=a+1.0;
3789 qam=a-1.0;
3790 bz=1.0-qab*x/qap;
3791 for (m=1;m<=taMath_float::max_iterations;m++) {
3792 em=(float) m;
3793 tem=em+em;
3794 d=em*(b-em)*x/((qam+tem)*(a+tem));
3795 ap=az+d*am;
3796 bp=bz+d*bm;
3797 d = -(a+em)*(qab+em)*x/((qap+tem)*(a+tem));
3798 app=ap+d*az;
3799 bpp=bp+d*bz;
3800 aold=az;
3801 am=ap/bpp;
3802 bm=bp/bpp;
3803 az=app/bpp;
3804 bz=1.0;
3805 if (fabs(az-aold) < (taMath_float::err_tolerance*fabs(az))) return az;
3806 }
3807 fprintf(stderrstderr, "a or b too big, or max_iterations too small in betacf()\n");
