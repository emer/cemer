587 Layer::AccessMode lve_acc_md, int lve_gpidx,

591 LeabraUnit* lviu = (LeabraUnit*)lvi_lay->UnitAccess(lvi_acc_md, 0, lvi_gpidx);

593 float lvd = lveu->act_eq - MAX(lviu->act_eq, lv.min_lvi)(((lviu->act_eq) > (lv.min_lvi)) ? (lviu->act_eq) : (
lv.min_lvi));
594 if(lv.lvi_scale_min && (lvd < 0.0f) && (lviu->act_eq < lv.min_lvi)) {

608float LVeLayerSpec::Compute_LVDa(LeabraLayer* lve_lay, LeabraLayer* lvi_lay,
609 LeabraNetwork* net) {
610 float lv_da = 0.0f;
611 if(lve_lay->unit_groups && (lve_lay->gp_geom.n == lvi_lay->gp_geom.n)) {
612 for(int g=0; g<lve_lay->gp_geom.n; g++) {
613 lv_da += Compute_LVDa_ugp(lve_lay, lvi_lay, Layer::ACC_GP, g, Layer::ACC_GP, g, net);
614 }
615 lv_da /= (float)lve_lay->gp_geom.n; // average!
616 }
617 else if(lve_lay->gp_geom.n > 1) {
618 // one lvi and multiple lve's
619 for(int g=0; g<lve_lay->gp_geom.n; g++) {
620 lv_da += Compute_LVDa_ugp(lve_lay, lvi_lay, Layer::ACC_GP, g, Layer::ACC_LAY, 0, net);
621 }
622 lv_da /= (float)lve_lay->gp_geom.n; // average!
623 }
624 else if(lvi_lay->gp_geom.n > 1) {
625 // one lve and multiple lvi's
626 for(int g=0; g<lvi_lay->gp_geom.n; g++) {
627 lv_da += Compute_LVDa_ugp(lve_lay, lvi_lay, Layer::ACC_LAY, 0, Layer::ACC_GP, g, net);
628 }
629 lv_da /= (float)lvi_lay->gp_geom.n; // average!
630 }
631 else {
632 lv_da = Compute_LVDa_ugp(lve_lay, lvi_lay, Layer::ACC_LAY, 0, Layer::ACC_LAY, 0, net);
633 }
634 return lv_da;
